59matrix ComponentSpringArm::GetPutOnMatrix() const {
60 float trans[3] = {0, 0, 0};
61 float rot[3] = {spring_arm_rotate_.x, spring_arm_rotate_.y, spring_arm_rotate_.z};
62 float scale[3] = {1.0f, 1.0f, 1.0f};
63 matrix rmat;
64 RecomposeMatrixFromComponents(trans, rot, scale, rmat.f32_128_0);
65
66 matrix tmat = matrix::translate(0, 0, spring_arm_length_);
67
68 matrix mat = matrix::identity();
69 if(auto object = object_.lock()) {
70 if(auto transform = object->GetComponent<ComponentTransform>()) {
71 mat = transform->GetMatrix();
72 }
73 }
74 mat = mul(rmat, mat);
75 mat = mul(tmat, mat);
76
77 return mat;
78}
