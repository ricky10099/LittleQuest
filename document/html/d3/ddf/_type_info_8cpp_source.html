104//---------------------------------------------------------------------------
105//! 名前を指定して指定基底クラス型でnewする
106//---------------------------------------------------------------------------
107void* CreateInstanceFromName(std::string_view class_name, Type& base_type) {
108 Type* p = base_type.child();
109 bool returnFromTraverse = false;
110 Type* next = nullptr;
111
112 //----------------------------------------------------------
113 // 継承ツリー構造を探索
114 // スタック再帰を使わない高速なツリー探索 (stackless tree traversal)
115 //----------------------------------------------------------
116 while(p != &base_type) {
117 if(!returnFromTraverse) {
118 // 名前チェックして一致したら作成
119 if(p->className() == class_name) {
120 return p->createInstance();
121 }
122 }
123
124 if(p->child() && !returnFromTraverse) {
125 // 子がある場合は子を先に調べる。(子から探索で戻ってきた場合は除外)
126 next = p->child();
127 returnFromTraverse = false;
128 } else if(p->siblings()) {
129 // 兄弟がいる場合は兄弟を調べる
130 next = p->siblings();
131 returnFromTraverse = false;
132 } else {
133 // 親へ戻る。
134 next = p->parent();
135 returnFromTraverse = true;
136 }
137 p = next;
138 }
139
140 return nullptr;
141}
