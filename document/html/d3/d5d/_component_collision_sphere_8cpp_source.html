29void ComponentCollisionSphere::Draw() {
30 __super::Draw();
31
32 if(!Scene::IsEdit() && !collision_status_.is(CollisionBit::ShowInGame))
33 return;
34
35 float scale = 1.0f;
36 auto trans = GetWorldMatrix();
37
38 if(attach_node_ < 0) {
39 // モデルアタッチしてない場合のみサイズ変更
40 float sx = length(trans.axisVectorX());
41 float sy = length(trans.axisVectorY());
42 float sz = length(trans.axisVectorZ());
43 scale = (sx + sy + sz) / 3.0f;
44 }
45
46 SetUseLighting(FALSE);
47 SetLightEnable(FALSE);
48 DrawSphere3D(cast(trans.translate()), radius_ * scale, 10, GetColor(0, 255, 0), GetColor(0, 0, 0), FALSE);
49 SetLightEnable(TRUE);
50 SetUseLighting(TRUE);
51}

139//! @brief ワールドMatrixの取得
140//! @return 他のコンポーネントも含めた位置
141const matrix ComponentCollisionSphere::GetWorldMatrix() const {
142 auto transform = collision_transform_;
143
144 auto obj = GetOwner();
145 if(attach_node_ >= 0) {
146 auto mdl = obj->GetComponent<ComponentModel>();
147 if(mdl) {
148 transform = mul(transform, attach_node_matrix_);
149 }
150 } else {
151 auto cmp = obj->GetComponent<ComponentTransform>();
152 if(cmp) {
153 transform = mul(transform, cmp->GetWorldMatrix());
154 }
155 }
156
157 return transform;
158}
