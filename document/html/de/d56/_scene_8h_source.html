646//! @brief オブジェクト取得
647//! @tparam T 取得したいオブジェクトクラス
648//! @return 取得オブジェクト
649template<class T>
650std::shared_ptr<T> Scene::Base::GetObjectPtr(const std::string_view name) {
651 if(name.empty()) {
652 for(auto& obj: objects_) {
653 auto cast = std::dynamic_pointer_cast<T>(obj);
654
655 if(cast)
656 return cast;
657 }
658 } else {
659 for(auto& obj: objects_) {
660 if(name.compare(obj->GetNameDefault()) == 0) {
661 auto cast = std::dynamic_pointer_cast<T>(obj);
662
663 if(cast)
664 return cast;
665 }
666 }
667 for(auto& obj: objects_) {
668 if(name.compare(obj->GetName()) == 0) {
669 auto cast = std::dynamic_pointer_cast<T>(obj);
670
671 if(cast)
672 return cast;
673 }
674 }
675
676 // 作成前Objectも検査する
677 for(auto& obj: pre_objects_) {
678 if(name.compare(obj->GetNameDefault()) == 0) {
679 auto cast = std::dynamic_pointer_cast<T>(obj);
680
681 if(cast)
682 return cast;
683 }
684 }
685 for(auto& obj: pre_objects_) {
686 if(name.compare(obj->GetName()) == 0) {
687 auto cast = std::dynamic_pointer_cast<T>(obj);
688
689 if(cast)
690 return cast;
691 }
692 }
693 }
694
695 return nullptr;
696}
