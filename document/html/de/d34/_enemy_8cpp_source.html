41void Enemy::Update() {
42 float deltaTime = GetDeltaTime();
43
44 if(isDead) {
45 destroyTimer -= deltaTime;
46 return;
47 }
48
49 if(state != EnemyState::ATTACK && state != EnemyState::GET_HIT) {
50 if(!FindPlayer() && state != EnemyState::GIVE_UP && state != EnemyState::WAIT) {
51 if(!patrolPoint.empty()) {
52 ChangeState(EnemyState::PATROL);
53 } else {
54 ChangeState(EnemyState::IDLE);
55 }
56 }
57 }
58
59 float3 move;
60
61 switch(state) {
62 case EnemyState::GET_HIT:
63 if(!pModel.lock()->IsPlaying()) {
64 ChangeState(EnemyState::IDLE);
65 //isBusy = false;
66 }
67 break;
68 case EnemyState::GIVE_UP:
69 BackToInitialPosition(move);
70 break;
71 case EnemyState::CHASING:
72 ChasePlayer(move);
73 break;
74 case EnemyState::ATTACK:
75 Attack();
76 break;
77 case EnemyState::WAIT:
78 Waiting(deltaTime);
79 break;
80 case EnemyState::PATROL:
81 Patrol(move);
82 break;
83 case EnemyState::IDLE:
84 Idle();
85 break;
86 }
87
88 move *= speedBase * speedFactor * GetDeltaTime60();
89 AddTranslate(move);
90}
