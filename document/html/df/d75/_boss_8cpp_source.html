126void Boss::GameAction() {
127 if(m_damageTimer > 0) {
128 m_damageTimer -= GetDeltaTime60();
129 } else {
130 m_damageCount = 0;
131 }
132
133 if(m_bossCombo == BossCombo::NONE && m_state == BossState::IDLE) {
134 if(!m_isAngry) {
135 SelectAction();
136 } else {
137 SelectAngryAction();
138 }
139
140 if(m_pHP.lock()->GetHPRate() < 50.0f && !m_isAngry) {
141 ChangeState(BossState::ANGRY);
142 }
143 }
144 switch(m_state) {
145 case BossState::IDLE:
146 Idle();
147 break;
148 case BossState::ATTACK:
149 Attack();
150 break;
151 case BossState::WAIT:
152 Wait();
153 break;
154 case BossState::TURN_LEFT:
155 m_pModel.lock()->PlayAnimationNoSame(STR(BossState::TURN_LEFT));
156 if(!m_pModel.lock()->IsPlaying()) {
157 ChangeState(BossState::WAIT);
158 } else {
159 SetRotationToPositionWithLimit(m_pPlayer.lock()->GetTranslate(), 1);
160 }
161 break;
162 case BossState::TURN_RIGHT:
163 m_pModel.lock()->PlayAnimationNoSame(STR(BossState::TURN_RIGHT));
164 if(!m_pModel.lock()->IsPlaying()) {
165 ChangeState(BossState::WAIT);
166 } else {
167 SetRotationToPositionWithLimit(m_pPlayer.lock()->GetTranslate(), 1);
168 }
169 break;
170 case BossState::CHASE:
171 ChasePlayer();
172 break;
173 case BossState::ANGRY:
174 PowerUp();
175 break;
176 case BossState::GET_HIT:
177 Damaging();
178 break;
179 case BossState::DEAD:
180 Die();
181 break;
182 }
183}

260void Boss::SelectAction() {
261 float distance = GetDistance(this->GetTranslate(), m_pPlayer.lock()->GetTranslate());
262 float angle = GetDegreeToPosition(m_pPlayer.lock()->GetTranslate());
263 int random = rand() % 100 + 1;
264 float3 front = GetMatrix().axisZ() - GetTranslate();
265 float dotPlayer = VDot(cast(front), cast(m_pPlayer.lock()->GetTranslate()));
266
267 if(angle >= FRONT_ANGLE && angle <= BACK_ANGLE) {
268 if(dotPlayer < 0) {
269 ChangeState(BossState::TURN_RIGHT);
270 } else {
271 ChangeState(BossState::TURN_LEFT);
272 }
273 } else if(distance < TOO_CLOSE_DISTANCE && angle < FRONT_ANGLE) {
274 ChangeState(BossState::ATTACK);
275 if(random <= 35) {
276 m_bossCombo = BossCombo::BACKFLIP_PUNCH;
277 } else if(random <= 70) {
278 SetRotationToPositionWithLimit(m_pPlayer.lock()->GetTranslate(), 30);
279 m_bossCombo = BossCombo::SWIP;
280 } else if(random <= 80) {
281 SetRotationToPositionWithLimit(m_pPlayer.lock()->GetTranslate(), 50);
282 m_bossCombo = BossCombo::COMBO5;
283 } else {
284 if(dotPlayer < 0) {
285 ChangeState(BossState::TURN_RIGHT);
286 } else {
287 ChangeState(BossState::TURN_LEFT);
288 }
289 }
290 m_combo = 1;
291 } else if(distance < CLOSE_DISTANCE && angle < FRONT_ANGLE) {
292 ChangeState(BossState::ATTACK);
293 if(random <= 30) {
294 SetRotationToPositionWithLimit(m_pPlayer.lock()->GetTranslate(), 50);
295 m_bossCombo = BossCombo::COMBO5;
296 } else if(random <= 90) {
297 SetRotationToPositionWithLimit(m_pPlayer.lock()->GetTranslate(), 30);
298 m_bossCombo = BossCombo::SWIP;
299 } else {
300 if(dotPlayer < 0) {
301 ChangeState(BossState::TURN_RIGHT);
302 } else {
303 ChangeState(BossState::TURN_LEFT);
304 }
305 }
306 m_combo = 1;
307 } else if(distance < MIDDLE_DISTANCE && angle > BACK_ANGLE) {
308 if(random <= 30) {
309 ChangeState(BossState::ATTACK);
310 m_bossCombo = BossCombo::BACKFLIP_PUNCH;
311 m_combo = 1;
312 } else {
313 if(dotPlayer < 0) {
314 ChangeState(BossState::TURN_RIGHT);
315 } else {
316 ChangeState(BossState::TURN_LEFT);
317 }
318 }
319 } else if(distance >= CLOSE_DISTANCE && distance <= FAR_DISTANCE) {
320 ChangeState(BossState::CHASE);
321 } else if(distance > FAR_DISTANCE) {
322 ChangeState(BossState::ATTACK);
323 SetRotationToPosition(m_pPlayer.lock()->GetTranslate());
324 m_bossCombo = BossCombo::CHARGE_PUNCH;
325 m_combo = 1;
326 }

412void Boss::AttackAnimation(std::string animName, AnimInfo& animInfo, std::vector<ComponentCollisionCapsulePtr> atkCol,
413 bool playSE) {
414 if(m_pModel.lock()->GetPlayAnimationName() != animName) {
415 m_pModel.lock()->PlayAnimationNoSame(animName, false, 0.2F, animInfo.animStartTime);
416 m_pModel.lock()->SetAnimationSpeed(animInfo.animStartSpeed);
417 m_playedSE = false;
418 }
419 m_currAnimTime = m_pModel.lock()->GetAnimationPlayTime();
420 if(m_currAnimTime >= animInfo.triggerStartTime) {
421 m_pModel.lock()->SetAnimationSpeed(animInfo.animSpeed);
422 for(int i = 0; i < atkCol.size(); i++) {
423 atkCol[i]->SetHitCollisionGroup((u32)ComponentCollision::CollisionGroup::PLAYER);
424 }
425
426 if(!m_playedSE && playSE) {
427 PlaySoundMem(m_attackSE, DX_PLAYTYPE_BACK);
428 ChangeVolumeSoundMem((int)(MAX_VOLUME * (Scene::GetSEVolume() / 100.0f)), m_attackSE);
429 m_playedSE = true;
430 }
431 }
432 if(m_currAnimTime >= animInfo.triggerEndTime) {
433 for(int i = 0; i < atkCol.size(); i++) {
434 atkCol[i]->SetHitCollisionGroup((u32)ComponentCollision::CollisionGroup::NONE);
435 }
436 }
437 if(m_currAnimTime >= animInfo.animCutInTime) {
438 m_combo++;
439 m_isHitPlayer = false;
440 }
441}
