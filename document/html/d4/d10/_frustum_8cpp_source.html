26//---------------------------------------------------------------------------
27//! 更新
28//---------------------------------------------------------------------------
29void Frustum::update() {
30 //----------------------------------------------------------
31 // 行列を作成
32 //----------------------------------------------------------
33 // ビュー行列
34 mat_view_ = matrix::lookAtLH(position_, look_at_, world_up_);
35
36 // カメラのワールド行列はビュー行列の逆行列
37 mat_camera_world_ = inverse(mat_view_);
38
39 // 投影行列
40 switch(depth_mode_) {
41 case DepthMode::Default:
42 mat_proj_ = matrix::perspectiveFovLH(fovy_, aspect_ratio_, z_near_, z_far_);
43 break;
44 case DepthMode::Reverse:
45 // farとnearを入れ替えることでreverse-zになる
46 mat_proj_ = matrix::perspectiveFovLH(fovy_, aspect_ratio_, z_far_, z_near_);
47 break;
48 case DepthMode::ReverseInfinite:
49 // 無限遠まで高精度レンダリングする特殊投影行列
50 mat_proj_ = matrix::perspectiveFovInfiniteFarPlaneLH(fovy_, aspect_ratio_, z_near_);
51 break;
52 }
53
54 // 合成
55 mat_view_proj_ = mul(mat_view_, mat_proj_);
56}

61void Frustum::renderDebug() {
62 f32 zf = 10.0f; // 近クリップ距離
63 f32 zn = 1.0f; // 遠クリップ距離
64
65 // nearとfarの距離の矩形の幅・高さ
66 f32 nw = tanf(fovy_ * 0.5f) * zn * aspect_ratio_;
67 f32 nh = tanf(fovy_ * 0.5f) * zn;
68 f32 fw = tanf(fovy_ * 0.5f) * zf * aspect_ratio_;
69 f32 fh = tanf(fovy_ * 0.5f) * zf;
70
71 // カメラの姿勢情報
72 float3 axis_x = mat_camera_world_.axisX();
73 float3 axis_y = mat_camera_world_.axisY();
74 float3 axis_z = mat_camera_world_.axisZ();
75
76 //----------------------------------------------------------
77 // 前後四隅の点を計算
78 //----------------------------------------------------------
79 float3 v[8];
80 v[0] = position_ + axis_z * zn - axis_x * nw + axis_y * nh;
81 v[1] = position_ + axis_z * zn + axis_x * nw + axis_y * nh;
82 v[2] = position_ + axis_z * zn + axis_x * nw - axis_y * nh;
83 v[3] = position_ + axis_z * zn - axis_x * nw - axis_y * nh;
84 v[4] = position_ + axis_z * zf - axis_x * fw + axis_y * fh;
85 v[5] = position_ + axis_z * zf + axis_x * fw + axis_y * fh;
86 v[6] = position_ + axis_z * zf + axis_x * fw - axis_y * fh;
87 v[7] = position_ + axis_z * zf - axis_x * fw - axis_y * fh;
88
89 SetUseLighting(false);
90
91 //----------------------------------------------------------
92 // デプス無効でシルエットを描画
93 //----------------------------------------------------------
94 SetUseZBuffer3D(false);
95 {
96 DrawLine3D(cast(v[0]), cast(v[1]), GetColor(0, 0, 0));
97 DrawLine3D(cast(v[1]), cast(v[2]), GetColor(0, 0, 0));
98 DrawLine3D(cast(v[2]), cast(v[3]), GetColor(0, 0, 0));
99 DrawLine3D(cast(v[3]), cast(v[0]), GetColor(0, 0, 0));
100
101 DrawLine3D(cast(v[4]), cast(v[5]), GetColor(0, 0, 0));
102 DrawLine3D(cast(v[5]), cast(v[6]), GetColor(0, 0, 0));
103 DrawLine3D(cast(v[6]), cast(v[7]), GetColor(0, 0, 0));
104 DrawLine3D(cast(v[7]), cast(v[4]), GetColor(0, 0, 0));
105
106 DrawLine3D(cast(v[0]), cast(v[4]), GetColor(0, 0, 0));
107 DrawLine3D(cast(v[1]), cast(v[5]), GetColor(0, 0, 0));
108 DrawLine3D(cast(v[2]), cast(v[6]), GetColor(0, 0, 0));
109 DrawLine3D(cast(v[3]), cast(v[7]), GetColor(0, 0, 0));
110
111 DrawLine3D(cast(position_ + axis_z * zn), cast(position_ + axis_z * zf * 2.0f), GetColor(0, 0, 0));
112 }
113
114 for(u32 i = 0; i < 8; ++i) {
115 DrawSphere3D(cast(v[i]), 0.1f, 8, GetColor(255, 255, 0), 0, true);
116 }
117
118 //----------------------------------------------------------
119 // デプス有効でシルエットを描画
120 //----------------------------------------------------------
121 SetUseZBuffer3D(true);
122 {
123 DrawLine3D(cast(v[0]), cast(v[1]), GetColor(0, 255, 0));
124 DrawLine3D(cast(v[1]), cast(v[2]), GetColor(0, 255, 0));
125 DrawLine3D(cast(v[2]), cast(v[3]), GetColor(0, 255, 0));
126 DrawLine3D(cast(v[3]), cast(v[0]), GetColor(0, 255, 0));
127
128 DrawLine3D(cast(v[4]), cast(v[5]), GetColor(0, 255, 0));
129 DrawLine3D(cast(v[5]), cast(v[6]), GetColor(0, 255, 0));
130 DrawLine3D(cast(v[6]), cast(v[7]), GetColor(0, 255, 0));
131 DrawLine3D(cast(v[7]), cast(v[4]), GetColor(0, 255, 0));
132
133 DrawLine3D(cast(v[0]), cast(v[4]), GetColor(0, 255, 0));
134 DrawLine3D(cast(v[1]), cast(v[5]), GetColor(0, 255, 0));
135 DrawLine3D(cast(v[2]), cast(v[6]), GetColor(0, 255, 0));
136 DrawLine3D(cast(v[3]), cast(v[7]), GetColor(0, 255, 0));
137
138 DrawLine3D(cast(position_ + axis_z * zn), cast(position_ + axis_z * zf * 2.0f), GetColor(0, 255, 0));
139 }
140 SetUseLighting(true);
141}

269//! スクリーンまでの距離を取得
270//---------------------------------------------------------------------------
271f32 Frustum::screenDistance(u32 resolution_height) const {
272 // +
273 // ／ |
274 // ／ |
275 // ／ | resolution_height /2
276 // ／ |
277 // ／）fovy_ / 2 |
278 //+-----------------+
279 // d スクリーンまでの距離
280 //
281 // tan(fovy_ / 2) = (resolution_height / 2) / d
282 //
283 return static_cast<f32>(resolution_height / 2) / tanf(fovy_ * 0.5f);
284}

303float3 Frustum::rayFromScreenPosition(float2 screen_position) {
304 // カメラの姿勢情報
305 float3 axis_x = mat_camera_world_.axisX();
306 float3 axis_y = mat_camera_world_.axisY();
307 float3 axis_z = mat_camera_world_.axisZ();
308
309 return screen_position.x * axis_x + screen_position.y / aspect_ratio_ * axis_y + axis_z;
310}
