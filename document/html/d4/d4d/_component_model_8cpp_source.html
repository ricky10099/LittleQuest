13void ComponentModel::Load(std::string_view path) {
14 model_status_.off(ModelBit::ErrorFileNotFound);
15 model_status_.on(ModelBit::SetFileAlready);
16
17 path_ = path;
18
19 // ファイルが存在しているか?
20 if(!HelperLib::File::CheckFileExistence(path_)) {
21 // ロードできなかった
22 model_status_.on(ModelBit::ErrorFileNotFound);
23 model_status_.off(ModelBit::Initialized);
24 nodes_name_.clear();

27
28 bool result = model_->load(path_);
29 if(!result) {
30 // ロードできなかった
31 model_status_.on(ModelBit::ErrorFileNotFound);
32 } else {
33 // ロード済み
34 model_status_.on(ModelBit::Initialized);
35 }
36
37 // ノード名を初期化しておく
38 nodes_name_.clear();

41//! @brief モデル更新
42//! @param delta 1フレームの秒数
43void ComponentModel::Update() {
44 float delta = GetDeltaTime();
45
46 // モデルが存在しているならばTransform設定を行う
47 if(IsValid()) {
48 // アニメーションがあり再生している?
49 if(animation_ && animation_->isValid() && animation_->isPlaying()) {
50 animation_->update(delta);
51 animation_time_ += delta;
52
53 // 進めた結果終了した場合はOldとして確保
54 if(!animation_->isPlaying()) {
55 old_animation_name_ = current_animation_name_;
56 }
57 }
58
59 auto mat = model_transform_;
60 auto trns = GetOwner()->GetComponent<ComponentTransform>();
61 if(trns) {
62 mat = mul(mat, trns->GetWorldMatrix());
63 }
64
65 if(model_) {
66 // ワールド行列を設定
67 model_->setWorldMatrix(mat);
68
69 // モデル更新
70 model_->update(delta);
71 }
72 }
73}
