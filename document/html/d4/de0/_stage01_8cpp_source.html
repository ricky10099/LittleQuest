236//! @param [in] delta 経過時間
237//---------------------------------------------------------------------------
238void Stage01::Update() {
239 float t;
240 float3 newCamPos;
241 float3 newCamTarget;
242 float newFOV;
243
244#ifndef _DEBUG
245 HideMouse(true);
246#endif // !_DEBUG
247
248 switch(scene_state) {
249 case Scene::SceneState::TRANS_IN:
250 if(FadeIn()) {
251 m_pBoss.lock()->PlayTaunt();
252 }
253
254 if(m_pBoss.lock()->IsPlayedTaunt()) {
255 m_cutSceneTimer -= GetDeltaTime60();
256 m_cutSceneTimer = std::max(0.0f, m_cutSceneTimer);
257 t = abs(1 - (m_cutSceneTimer / START_CUT_SCENE_TIME));
258 newCamPos = lerp(CUT_SCENE_POS_1, m_pPlayerCamera.lock()->GetTranslate(), t);
259 newCamTarget =
260 lerp(m_pBoss.lock()->GetTranslate() + float3{0, 20, 0}, m_pPlayer.lock()->GetTranslate() + float3{0, 5, 0}, t);
261 newFOV = lerp(float1(FOV_INTRO), FOV_ORG, t);
262 m_pCamera.lock()->SetPositionAndTarget(newCamPos, newCamTarget);
263 m_pCamera.lock()->SetPerspective(newFOV);
264 }
265
266 if(m_cutSceneTimer <= 0) {
267 m_pPlayerCamera.lock()->SetCurrentCamera();
268 scene_state = Scene::SceneState::GAME;
269 m_pPlayer.lock()->SetSceneState(scene_state);
270 m_pBoss.lock()->SetSceneState(scene_state);
271 }
272
273 if(IsKeyDown(KEY_INPUT_RETURN) || IsMouseDown(MOUSE_INPUT_1) || IsKeyDown(KEY_INPUT_SPACE) ||
274 /*IsPadDown(PAD_ID::PAD_10, DX_PADTYPE_DUAL_SENSE) || */ IsPadDown(PAD_ID::PAD_3, DX_PADTYPE_DUAL_SENSE)) {
275 m_fadeTimer = 0;
276 m_alpha = 0;
277 m_cutSceneTimer = 0;
278 m_pCamera.lock()->SetPerspective(FOV_ORG);
279 }
280 break;
281 case Scene::SceneState::GAME:
282 m_second -= GetDeltaTime();
283 if(m_second <= 0) {
284 if(m_minute <= 0) {
285 m_isLose = true;
286 m_second = 0.0f;
287 } else {
288 m_second = 59.99f;
289 }
290 m_minute--;
291 m_minute = std::max(0, m_minute);
292 }
293 if((m_pBoss.lock()->IsDead() || m_pPlayer.lock()->IsDead() || m_isLose) && FadeOut()) {
294 scene_state = Scene::SceneState::TRANS_OUT;
295 m_pPlayer.lock()->SetSceneState(scene_state);
296 m_pBoss.lock()->SetSceneState(scene_state);
297 m_pPlayer.lock()->SetTranslate(PLAYER_SPAWN_POS);
298 m_pBoss.lock()->SetTranslate(BOSS_SPAWN_POS);
299 m_pBoss.lock()->SetRotationAxisXYZ({0, 90, 0});
300 }
301 break;
302 case Scene::SceneState::TRANS_OUT:
303 Scene::SetCanPause(false);
304 m_endingTimer -= GetDeltaTime60();
305
306 StopSoundMem(m_BGM);
307 StopSoundMem(m_introBGM);
308 m_pBoss.lock()->PlayDead();
309 m_pPlayer.lock()->PlayDead();
310
311 m_pCamera.lock()->SetCurrentCamera();
312 if(m_pBoss.lock()->IsDead()) {
313 m_pCamera.lock()->SetPositionAndTarget(BOSS_DEATH_CAM, m_pBoss.lock()->GetTranslate() + float3{0, 15, 0});
314 m_pShowImage = m_pClearImage.lock();
315 } else {
316 m_pCamera.lock()->SetPositionAndTarget(PLAYER_DEATH_CAM, m_pPlayer.lock()->GetTranslate() + float3{0, 10, 0});
317 m_pShowImage = m_pFailImage.lock();
318 PlaySoundMem(m_loseAudio, DX_PLAYTYPE_BACK);
319 }
320 break;
321 }
322}

328void Stage01::LateDraw() {
329 int screen_width, screen_height;
330 GetScreenState(&screen_width, &screen_height, NULL);
331
332 SetDrawBlendMode(DX_BLENDMODE_ALPHA, (int)m_alpha);
333 DrawBox(0, 0, screen_width, screen_height, 0u, TRUE);
334 SetDrawBlendMode(DX_BLENDMODE_NOBLEND, NULL);
335 int timerColor = 0xffffff;
336 if(m_minute <= 0) {
337 timerColor = 0xff0000;
338 }
339
340 switch(scene_state) {
341 case Scene::SceneState::TRANS_IN:
342 break;
343 case Scene::SceneState::GAME:
344 DrawFormatStringToHandle((int)((screen_width * 0.9f) - (m_stringWidth * 0.5f)), (int)(screen_height * 0.1), timerColor,
345 m_timerFontHandle, "%02i:%06.3f", m_minute, m_second);
346 break;
347 case Scene::SceneState::TRANS_OUT:
348 if(FadeIn()) {
349 if(ShowMessage()) {
350 DrawStringToHandle((int)((screen_width * 0.5f) - (m_stringWidth * 0.5f)), (int)(screen_height * 0.8),
351 "Press any key back to Title", 0xffee42, m_fontHandle, 0xffaf3f);
352 if(IsKeyDown(KEY_INPUT_RETURN) || IsMouseDown(MOUSE_INPUT_1) || IsKeyDown(KEY_INPUT_SPACE) ||
353 IsKeyDown(KEY_INPUT_ESCAPE) || IsPadDown(PAD_ID::PAD_10, DX_PADTYPE_DUAL_SENSE) ||
354 IsPadDown(PAD_ID::PAD_3, DX_PADTYPE_DUAL_SENSE) || m_endingTimer <= 0) {
355 Scene::Change(Scene::GetScene<GameTitleScene>());
356 }
357 } else {
358 if(IsKeyDown(KEY_INPUT_RETURN) || IsMouseDown(MOUSE_INPUT_1) || IsKeyDown(KEY_INPUT_SPACE) ||
359 IsKeyDown(KEY_INPUT_ESCAPE) || IsPadDown(PAD_ID::PAD_10, DX_PADTYPE_DUAL_SENSE) ||
360 IsPadDown(PAD_ID::PAD_3, DX_PADTYPE_DUAL_SENSE)) {
361 m_shrinkTimer = SHRINK_TIME;
362 }
363 }
364 m_pShowImage.lock()->SetPosition(
365 (screen_width * 0.2f), (screen_height * (0.0f + (0.4f * (m_shrinkTimer / SHRINK_TIME)))), (screen_width * 0.8f),
366 (screen_height * (1.0f - (0.4f * (m_shrinkTimer / SHRINK_TIME)))));
367 m_pShowImage.lock()->DrawTexture();
368 }
369 break;
370 }
