57void ComponentTargetTracking::PostUpdate() {
58 // アニメーション適応後、
59 // tracked_node_index_からノードマトリクスを取得してそれに対して
60 // さらなる向きの方向(now_xy_)を適応してTrackingを行うように変更する
61
62 if(tracked_node_index_ < 0)
63 return;
64
65 float3 target_pos = look_at_;
66
67 if(auto target = tracking_object_.lock())
68 target_pos = target->GetTranslate();
69
70 // Objectのモデルを先にWeakPtrで取得して確保しておく
71 if(owner_model_.lock() == nullptr)
72 owner_model_ = GetOwner()->GetComponent<ComponentModel>();
73
74 if(auto model = owner_model_.lock()) {
75 // trackingするマトリクス
76 tracking_matrix_ = cast(MV1GetFrameLocalMatrix(model->GetModel(), tracked_node_index_));
77 // 正しい向きのノードMatrixを取得しておく
78 float3 node_rot;
79 float3 node_trans;
80 float3 node_scale;
81 DecomposeMatrixToComponents((float*)tracking_matrix_.f32_128_0, (float*)&node_trans, (float*)&node_rot,
82 (float*)&node_scale);
83#if 1
84 // モデルの方向を取得する
85 float3 model_vec = normalize(mul(float4(front_vector_, 0), model->GetWorldMatrix()).xyz);
86
87 // モデルを通常に戻すインバーズマトリクスを取得しておく
88 auto fmat = inverse(HelperLib::Math::CreateMatrixByFrontVector(model_vec));
89
90 // ターゲットへの向きを取得して(
91 // MV1GetFrameLocalWorldMatrixでとると100%位置はあっているがどうやら…1Frame遅れる)
92 float3 node_pos = tracking_matrix_._41_42_43;
93 float3 model_pos = mul(float4(node_pos, 1), GetOwner()->GetMatrix()).xyz;
94
95 float3 vec = target_pos - model_pos;
96 auto mat = HelperLib::Math::CreateMatrixByFrontVector(vec, {0, 1, 0}, true);
97
98 // それにインバースを掛け合わせて0基準のベクトルに変換する
99 mat = mul(mat, fmat);
100 mat._41_42_43 = node_pos;
101
102 // 現状の回転と照らし合わせて回転を計算する
103 float3 new_rot;
104 float3 new_trans;
105 float3 new_scale;
106 // その回転をノードMatrixにさらに加えて向きの回転を実現する
107 DecomposeMatrixToComponents((float*)mat.f32_128_0, (float*)&new_trans, (float*)&new_rot, (float*)&new_scale);
108
109 // ZベクトルをY軸固定で角度に変化させる
110 float3 zvec = mat.axisZ();
111 float lenzx = sqrt((zvec.z * zvec.z) + (zvec.x * zvec.x));
112 float lenzy = sqrt((zvec.z * zvec.z) + (zvec.y * zvec.y));
113
114 new_rot.x = atan2((float)zvec.y, (float)lenzx) * RadToDeg;
115 new_rot.y = atan2((float)zvec.x, (float)lenzy) * RadToDeg;
116 new_rot.z = 0;
117
118 if(new_rot.y > 180)
119 new_rot.y = new_rot.y - 360;
120 if(new_rot.x < -180)
121 new_rot.x += 360;
122
123 // 右リミット
124 if((float)new_rot.y > (float)limit_lr_.y) {
125 new_rot.y = limit_lr_.y;
126 }
127 // 左リミット
128 if((float)new_rot.y < -(float)limit_lr_.x) {
129 new_rot.y = -limit_lr_.x;
130 }
131
132 // 上リミット
133 if((float)new_rot.x > (float)limit_ud_.x) {
134 new_rot.x = limit_ud_.x;
135 }
136 // 下リミット
137 if((float)new_rot.x < -(float)limit_ud_.y) {
138 new_rot.x = -limit_ud_.y;
139 }
140
141 float2 diff = {new_rot.x - now_rot_.x, new_rot.y - now_rot_.y};
142 float len = length(diff);
143 if(len > limit_frame_) {
144 diff = normalize(diff) * limit_frame_;
145 }
146
147 now_rot_.x += diff.x;
148 now_rot_.y += diff.y;
149
150 node_rot.x += now_rot_.x;
151 node_rot.y += now_rot_.y;
152
153#endif
154 RecomposeMatrixFromComponents((float*)&node_trans, (float*)&node_rot, (float*)&node_scale, (float*)mat.f32_128_0);
155
156 // auto roty = matrix::rotateY(radians((float1)theta_xz));
157 // auto rotx = matrix::rotateX(radians((float1)theta_yz));
158 // mat = mul(mul(tracking_matrix_, rotx), roty);
159
160 MV1SetFrameUserLocalMatrix(model->GetModel(), tracked_node_index_, cast(mat));
161
162 // これをONにすると他のGizmoのIsOverなどがチェックが取れずPICKが常に有効になる(地面が選ばれてしまう)
163 // ShowGizmo( (float*)mat.f32_128_0, ImGuizmo::OPERATION::TRANSLATE,
164 // ImGuizmo::MODE::LOCAL, 0xab1245 );
165 }

183void ComponentTargetTracking::GUI() {
184 assert(GetOwner());
185 auto obj_name = GetOwner()->GetName();
186
187 ImGui::Begin(obj_name.data());
188 {
189 ImGui::Separator();
190 if(ImGui::TreeNode(u8"TargetTracking")) {
191 if(ImGui::Button(u8"削除")) {
192 GetOwner()->RemoveComponent(shared_from_this());
193 }
194
195 int select_node_index = -1;
196 if(auto model = GetOwner()->GetComponent<ComponentModel>()) {
197 auto list = model->GetNodesName();
198 select_node_index = Combo(u8"トラッキングノード", tracked_node_, list);
199 if(select_node_index < list.size())
200 SetTrackingNode(select_node_index);
201 }
202
203 if(auto scene = Scene::GetCurrentScene()) {
204 std::string name = "";
205 if(auto obj = tracking_object_.lock()) {
206 name = obj->GetName();
207 }
208
209 auto objs = scene->GetObjectPtrVec();
210 std::vector<std::string_view> names;
211 names.push_back(null_name);
212 for(auto obj: objs) {
213 names.push_back(obj->GetName());
214 }
215 int id = Combo(u8"ターゲットオブジェクト", name, names) - 1;
216
217 if(id >= 0) {
218 tracking_object_ = objs[id];
219 } else if(id == -1) {
220 tracking_object_.reset();
221 }
222 }
223
224 tracking_status_.off(TrackingBit::ObjectTracking);
225 if(auto obj = tracking_object_.lock()) {
226 tracking_status_.on(TrackingBit::ObjectTracking);
227 }
228
229 if(!tracking_status_.is(TrackingBit::ObjectTracking)) {
230 ImGui::DragFloat3(u8"ターゲットポイント", (float*)&look_at_, 0.01f);
231 }
232
233 ImGui::Separator();
234 ImGui::DragFloat2(u8"左右角度制限", (float*)&limit_lr_, 0.1f, 0, 180, "%.1f");
235 ImGui::DragFloat2(u8"上下角度制限", (float*)&limit_ud_, 0.1f, 0, 89, "%.1f");
236 ImGui::Separator();
237 ImGui::DragFloat3(u8"前ベクトル", (float*)&front_vector_, 0.1f, 0, 180, "%.1f");
238
239 ImGui::TreePop();
240 }
241 }
242 ImGui::End();
