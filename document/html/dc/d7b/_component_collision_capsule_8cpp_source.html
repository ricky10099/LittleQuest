43void ComponentCollisionCapsule::PrePhysics() {
44#ifdef USE_JOLT_PHYSICS
45 __super::PrePhysics();
46
47 auto owner = GetOwner();
48 auto mat = mul(Matrix(), owner->GetMatrix());
49
50 if(update_delta_time_ <= 0.0f)
51 update_delta_time_ = 1.0f / 60.0f;
52
53 float3 vec = {0, 0, 0};
54 if(IsCollisionStatus(CollisionBit::UsePhysics)) {
55 // Physicsでの移動
56 vec = mat._41_42_43 - mul(Matrix(), owner->GetWorldMatrix())._41_42_43;
57 vec *= (1.0f / update_delta_time_);
58 } else {
59 // ワープ
60 character_->setPosition(mat._41_42_43);
61 // Physics移動に変更
62 SetCollisionStatus(CollisionBit::UsePhysics, true);
63 }
64# if 1
65 character_->setRotation( // rotation
66 quaternion( // quaternion {
67 {
68 {mat._11_12_13}, //
69 {mat._21_22_23}, //
70 {mat._31_32_33}, //
71 }) // } quaternion
72 );
73# endif
74 auto* physicsEngine = physics::Engine::instance();
75 // physicsEngine->setGravity( {0,-0.98,0} );
76
77 // OwnerのOldMatrixと現在のMatrixから移動量を割り出す
78 auto old_pos = GetOwner()->GetOldWorldMatrix().translate();
79 auto mov = GetOwner()->GetTranslate() - old_pos;
80
81 float3 new_velocity = mov * (1.0f / update_delta_time_);
82
83# if 1
84 // 重力
85 if(use_gravity_)
86 gravity_ += physicsEngine->gravity() * 0.1f; //< キャラが重力にて滑るため、0.1倍にしています
87
88 // プレイヤー操作
89 new_velocity += vec + gravity_;
90 // character_->setMass( 600.0f );
91
92 // 速度を更新
93 character_->setLinearVelocity(new_velocity);
94
95 character_->update(update_delta_time_);
96# endif
97#endif // USE_JOLT_PHYSICS
98}

121void ComponentCollisionCapsule::Draw() {
122#ifdef USE_JOLT_PHYSICS
123 if(character_ == nullptr)
124 return;
125
126 float half_height = (height_ - radius_ * 2) * 0.5f;
127
128 auto mat = matrix(character_->rotation());
129 mat = mul(mat, matrix::translate(float3{0, half_height + radius_, 0}));
130 mat = mul(mat, Matrix());
131
132 float3 pos = mul(float4(character_->position(), 1), mat).xyz;
133
134 auto top = pos + mat.axisY() * half_height;
135 auto bottom = pos - mat.axisY() * half_height;
136
137 SetUseLighting(FALSE);
138 SetLightEnable(FALSE);
139 DrawCapsule3D(cast(top), cast(bottom), radius_, 10, GetColor(0, 255, 0), GetColor(0, 0, 0), FALSE);
140 SetUseLighting(TRUE);
141 SetLightEnable(TRUE);
142#else
143 __super::Draw();
144
145 float scale = 1.0f;
146
147 auto trans = GetWorldMatrix();
148
149 if(attach_node_ < 0) {
150 // モデルアタッチしてない場合のみサイズ変更
151 float sx = length(trans.axisVectorX());
152 float sy = length(trans.axisVectorY());
153 float sz = length(trans.axisVectorZ());
154 scale = (sx + sy + sz) / 3.0f;
155 }
156
157 float radius = radius_ * scale;
158
159 SetUseLighting(FALSE);
160 SetLightEnable(FALSE);
161
162 // 半径に合わせて高さの再設定
163 if(height_ < radius * 2)
164 height_ = radius * 2;
165
166 if(!Scene::IsEdit() && !collision_status_.is(CollisionBit::ShowInGame))
167 return;
168
169 float3 pos2 = trans.translate();
170 float3 pos1 = normalize((float3&)trans.axisVectorY()) * height_ + pos2;
171
172 float3 vec = normalize(pos2 - pos1);
173
174 DrawCapsule3D(cast(pos1 + vec * radius), cast(pos2 - vec * radius), radius, 10, GetColor(0, 255, 0), GetColor(0, 0, 0),
175 FALSE);
176 SetLightEnable(TRUE);
177 SetUseLighting(TRUE);
178#endif
179}
