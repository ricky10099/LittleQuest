42//! 作成
43//---------------------------------------------------------------------------
44bool Animation::load(const Animation::Desc* desc, size_t desc_count) {
45 is_valid_ = true; // 先に一旦trueにする
46
47 // パラメーター初期化
48 for(u32 i = 0; i < desc_count; ++i) {
49 const auto& x = desc[i];
50
51 //------------------------------------------------------
52 // アニメーションリソース作成
53 // 既に同名ファイルがある場合は共有
54 //------------------------------------------------------
55 const std::string& resource_path = x.file_path_;
56
57 auto it = resource_pool.find(resource_path);
58 if(it == resource_pool.end()) { // 新規登録
59 resource_pool[resource_path] = std::make_shared<ResourceAnimation>(resource_path);
60 }
61
62 // 共有
63 std::shared_ptr<ResourceAnimation>& resource = resource_pool[resource_path];
64
65 descs_.push_back(x);
66 mv1_handles_.push_back(MV1DuplicateModel(*resource));
67
68 if(resource->isValid() == false) {
69 is_valid_ = false; // 一度でもエラーの場合はfalse
70 }
71
72 // 名前逆引きテーブルに登録
73 name_table_[desc[i].name_] = i;
74 }
75
76 return is_valid_;
77}

79//---------------------------------------------------------------------------
80//! 更新
81//---------------------------------------------------------------------------
82void Animation::update(f32 dt) {
83 assert(model_ && "アニメーションにモデルが設定されていません");
84
85 // 再生していない場合
86 if(isPlaying() == false)
87 return;
88
89 // ポーズ中の場合
90 if(is_paused_)
91 return;
92
93 //----------------------------------------------------------
94 // アニメーションを進める
95 //----------------------------------------------------------
96 for(u32 i = 0; i < 2; ++i) {
97 auto& c = contexts_[i];
98
99 if(c.animation_index_ == -1)
100 continue;
101
102 if(c.is_playing_ == false)
103 continue;
104
105 auto& desc = descs_[c.animation_index_];
106
107 // アニメーション再生時間を進める
108 c.play_time_ += dt * 30.0f * desc.animation_speed_;
109
110 if(c.is_loop_) {
111 // アニメーション再生時間がアニメーションの総時間を越えていたらループさせる
112 if(c.animation_total_time_ <= c.play_time_) {
113 c.play_time_ -= c.animation_total_time_;
114 }
115 } else {
116 // アニメーション再生時間がアニメーションの総時間を越えていたら停止
117 if(c.animation_total_time_ <= c.play_time_) {
118 c.play_time_ = c.animation_total_time_;
119 c.is_playing_ = false;
120 }
121 }
122
123 // 新しいアニメーション再生時間をセット
124 MV1SetAttachAnimTime(model_handle_, c.animation_attach_index_, c.play_time_);
125 }
126
127 //----------------------------------------------------------
128 // アニメーションブレンド
129 //----------------------------------------------------------
130
131 // ブレンドを進める
132 blend_ratio_ = std::max(0.0f, blend_ratio_ - 1.0f / blend_time_ * dt);
133
134 // 線形で等速補間すると硬い動きになるためEaseカーブ補間
135 auto ease = GetEaseFunction(EaseType::InOutCubic); // 加減速
136 f32 ratio = ease(blend_ratio_);
137
138 // ブレンド比率を設定
139 MV1SetAttachAnimBlendRate(model_handle_, contexts_[0].animation_attach_index_, 1.0f - ratio);
140 MV1SetAttachAnimBlendRate(model_handle_, contexts_[1].animation_attach_index_, ratio);
141
142 // 補間完了後は補間元のアニメーションを停止
143 if(blend_ratio_ == 0.0f) { // float値の==判定は0.0fのみ正確に判定できる
144 detachAnimation(1);
145 contexts_[1].is_playing_ = false;
146 }
147}

152void Animation::bindModel(Model* model) {
153 if(model == model_)
154 return;
155
156 // 旧モデルを解除
157 auto* last = model_;
158 if(last) {
159 // [DxLib]モデルとアニメーションの関連付けを解除
160 detachAnimation(0);
161 detachAnimation(1);
162
163 model_ = nullptr;
164 model_handle_ = -1;
165
166 last->bindAnimation(nullptr);
167 }
168
169 // 新モデルに登録
170 model_ = model;
171 if(model_) {
172 model_handle_ = *model_;
173 model_->bindAnimation(this);
174 }
175}

177//---------------------------------------------------------------------------
178//! アニメーションを再生する
179//---------------------------------------------------------------------------
180bool Animation::play(std::string_view name, bool is_loop, f32 blend_time, f32 start_time) {
181 // ブレンドの速度
182 blend_time_ = std::max(FLT_EPSILON, blend_time); // 0.0fにならないように
183
184 //----------------------------------------------------------
185 // [DxLib] 設定されているアニメーションを一旦解除
186 //----------------------------------------------------------
187 detachAnimation(0);
188 detachAnimation(1);
189
190 // 補間のために現在のアニメーションを補間前として設定
191 // [0] = 現在のアニメーション
192 // [1] = 以前のアニメーション
193 contexts_[1] = contexts_[0];
194
195 //----------------------------------------------------------
196 // 新規再生
197 //----------------------------------------------------------
198 {
199 // 名前からアニメーションを検索
200 std::string name_string(name);
201 auto it = name_table_.find(name_string);
202
203 if(it == name_table_.end()) {
204 return false; // 見つからなかった場合
205 }
206
207 auto& c = contexts_[0];
208
209 // アニメーション番号
210 c.animation_index_ = it->second;
211 // 現在の時間
212 c.play_time_ = start_time; // 開始位置は start_time から
213
214 c.is_playing_ = true;
215 c.is_loop_ = is_loop;
216 }
217
218 //----------------------------------------------------------
219 // アニメーション再生を設定
220 //----------------------------------------------------------
221 for(u32 i = 0; i < 2; ++i) {
222 attachAnimation(i);
223 }
224
225 // 補間対象のアニメーションがある場合は補間開始
226 if(contexts_[1].animation_index_ != -1) {
227 blend_ratio_ = 1.0f;
228 }
229
230 // ポーズ解除
231 is_paused_ = false;
232
233 return true;
234}

302//---------------------------------------------------------------------------
303bool Animation::attachAnimation(s32 context_index) {
304 auto& c = contexts_[context_index];
305
306 if(c.animation_index_ == -1)
307 return false;
308
309 auto& desc = descs_[c.animation_index_];
310
311 // モデルにアニメーションを設定
312 auto animation_index = desc.animation_index_; // アニメーション番号
313 auto mv1_handle = mv1_handles_[c.animation_index_]; // [DxLib] アニメーションのMV1
314
315 c.animation_attach_index_ = MV1AttachAnim(model_handle_, animation_index, mv1_handle, true);
316
317 // アニメーション総再生時間を取得
318 c.animation_total_time_ = MV1GetAttachAnimTotalTime(model_handle_, c.animation_attach_index_);
319
320 // アニメーション再生時間を初期化
321 MV1SetAttachAnimTime(model_handle_, c.animation_attach_index_, c.play_time_);
322
323 return true;
324}

341ResourceAnimation::ResourceAnimation(std::string_view path) {
342 //-----------------------------------------------------------------------
343 // MV1モデルアニメーションの読み込み
344 //-----------------------------------------------------------------------
345 std::string resource_path(path);
346
347 // パスの保存
348 path_ = convertTo(resource_path);
349
350 // アニメーションが含まれるモデルデータを読み込み
351 mv1_handle_ = MV1LoadModel(resource_path.c_str());
352
353 // ハンドルの非同期読み込み処理が完了したら呼ばれる関数
354 auto finish_callback = []([[maybe_unused]] int mv1_handle, void* data) {
355 auto* resource = reinterpret_cast<ResourceAnimation*>(data);
356
357 // アクティブフラグを設定
358 resource->active_ = true;
359 };
360
361 //----------------------------------------------------------
362 // 非同期読み込み
363 //----------------------------------------------------------
364 SetUseASyncLoadFlag(true);
365 {
366 // モデルの読み込み
367 mv1_handle_ = MV1LoadModel(resource_path.c_str());
368 SetASyncLoadFinishCallback(mv1_handle_, (void (*)(int, void*))finish_callback, this);
369 }
370 SetUseASyncLoadFlag(false);
371}
