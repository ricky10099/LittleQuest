273 //! @brief 押し戻し量の再計算
274 //! @param collision 相手のコリジョン
275 //! @param vec 自分100%の戻り量
276 //! @param push [out] 本当の自分の押し戻し量
277 //! @param other_push [out] 相手の押し戻し量
278 //! @return 押し戻しできる状態か?
279 bool CalcPush(ComponentCollisionPtr collision, float3 vec, float3* push, float3* other_push) const {
280 float mass1 = GetMass();
281 float mass2 = collision->GetMass();
282 if(mass1 < 0 && mass2 < 0) {
283 // 両方不動のものがぶつかった
284 *push = vec * 0.5f;
285 *other_push = -vec * 0.5f;
286 return false;
287 } else if(mass1 < 0) {
288 // 相手に跳ね返す
289 *push = {0, 0, 0};
290 *other_push = -vec;
291 } else if(mass2 < 0) {
292 // 相手に跳ね返す
293 *push = vec;
294 *other_push = {0, 0, 0};
295 } else {
296 // 重さに合わせて移動させる
297 mass1 += 0.001f;
298 mass2 += 0.001f;
299 float all = mass1 + mass2;
300 *push = vec * mass2 / all;
301 *other_push = -vec * mass1 / all;
302 }
303 return true;
304 }

306 bool IsGroupHit(ComponentCollisionPtr collision) const {
307 // コリジョンが当たらない設定になっている
308 if(collision_status_.is(CollisionBit::DisableHit) || collision->collision_status_.is(CollisionBit::DisableHit))
309 return false;
310
311 // 自分か相手のコリジョンタイプが不定
312 if(collision_type_ == CollisionType::NONE || collision->collision_type_ == CollisionType::NONE)
313 return false;
314
315 // 自分は相手のコリジョングループに当たらない
316 if(((u32)collision_group_ & collision->collision_hit_) == 0)
317 return false;
318
319 // 相手は自分のコリジョングループに当たらない
320 if(((u32)collision->collision_group_ & collision_hit_) == 0)
321 return false;
322
323 // 当たる設定となる
324 return true;
325 }
