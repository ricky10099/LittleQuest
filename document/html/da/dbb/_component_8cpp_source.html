114void Component::RegisterToObject(ComponentPtr cmp, ObjectPtr obj) {
115 // 同じタイプを許容しない
116 if(!cmp->GetStatus(StatusBit::SameType)) {
117 auto cmps = obj->GetComponents();
118 for(auto c: cmps) {
119 if(c->typeInfo()->className() == cmp->typeInfo()->className())
120 return;
121 }
122 }
123
124 cmp->Construct(obj);
125
126 obj->GetComponents().push_back(cmp);
127}
