74 //! プレイヤーをカメラを設定して追いかける時などに利用する(Default)
75 void SetUpdateTiming(ProcTiming timing) {
76 if(timing == ProcTiming::Update) {
77 camera_status_.on(CameraBit::UpdateOnUpdate);
78 camera_status_.off(CameraBit::UpdateOnPreUpdate);
79 camera_status_.off(CameraBit::UpdateOnPostUpdate);
80 } else if(timing == ProcTiming::PreUpdate) {
81 camera_status_.on(CameraBit::UpdateOnPreUpdate);
82 camera_status_.off(CameraBit::UpdateOnUpdate);
83 camera_status_.off(CameraBit::UpdateOnPostUpdate);
84 } else if(timing == ProcTiming::PostUpdate) {
85 camera_status_.on(CameraBit::UpdateOnPostUpdate);
86 camera_status_.off(CameraBit::UpdateOnPreUpdate);
87 camera_status_.off(CameraBit::UpdateOnUpdate);
