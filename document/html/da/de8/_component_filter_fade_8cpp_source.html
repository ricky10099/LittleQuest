43void ComponentFilterFade::Init() {
44 __super::Init();
45
46 auto draw = [&]() {
47 // 現在のRenderTargetを保存
48 auto target_desc = GetRenderTarget();
49 auto render_target = target_desc.color_targets_[0];
50 //------------------------------------------------------
51 // ワークテクスチャにコピー
52 //------------------------------------------------------
53 CopyToRenderTarget(texture_work_.get(), render_target);
54
55 //------------------------------------------------------
56 // シェーダー適用コピーして描き戻す
57 //------------------------------------------------------
58 SetWriteZBufferFlag(false); // フィルター用にデプスバッファ更新を無効化
59
60 // 定数バッファ更新
61 auto* p = reinterpret_cast<Parameter*>(GetBufferShaderConstantBuffer(parameter_cb_));
62 {
63 p->resolution_ = float4(WINDOW_W, WINDOW_H, 1.0f / WINDOW_W, 1.0f / WINDOW_H);
64 p->alpha_ = fade_alpha_;
65
66 // モザイク幅
67 f32 mosaic_factor = 1.0f - fade_alpha_; // フェードアウトすると粗くなっていく
68 f32 mosaic_width = mosaic_width_ * mosaic_factor;
69
70 p->mosaic_width_ = std::max(1u, static_cast<u32>(mosaic_width));
71
72 UpdateShaderConstantBuffer(parameter_cb_); // シェーダー用定数バッファハンドルの定数バッファへの変更を適用する
73 }
74
75 // 定数バッファをb4に設定
76 SetShaderConstantBuffer(parameter_cb_, DX_SHADERTYPE_PIXEL, 4);
77
78 // 全画面描画
79 CopyToRenderTarget(render_target, texture_work_.get(), *shader_ps_);
80
81 SetShaderConstantBuffer(-1, DX_SHADERTYPE_PIXEL, 4); // b4
82 SetWriteZBufferFlag(true); // 元に戻す
83
84 //------------------------------------------------------
85 // 元に戻す
86 //------------------------------------------------------
87 SetRenderTarget(target_desc);
88 };
89
90 // 描画タイミングを設定して描画
91 SetProc("FilterDraw", draw, ProcTiming::Filter, Priority::NORMAL);
92}

97void ComponentFilterFade::Update() {
98 f32 delta = GetDeltaTime();
99
100 if(finish_fade_)
101 return;
102
103 fade_alpha_ += fade_add_alpha_ * delta;
104 if(fade_alpha_ >= 1.0f) {
105 finish_fade_ = true;
106 fade_alpha_ = 1.0f;
107 }
108 if(fade_alpha_ < 0.0f) {
109 finish_fade_ = true;
110 fade_alpha_ = 0.0f;
111 }
112}
