55//! @return マトリクス
56matrix ComponentAttachModel::GetPutOnMatrix() const {
57 float trans[3] = {attach_model_offset_.x, attach_model_offset_.y, attach_model_offset_.z};
58 float rot[3] = {attach_model_rotate_.x, attach_model_rotate_.y, attach_model_rotate_.z};
59 float scale[3] = {1.0f, 1.0f, 1.0f};
60 matrix rmat;
61 RecomposeMatrixFromComponents(trans, rot, scale, rmat.f32_128_0);
62
63 matrix mat = matrix::identity();
64
65 if(auto object = object_.lock()) {
66 mat = object->GetMatrix();
67
68 if(auto target_model = object->GetComponent<ComponentModel>()) {
69 auto model_mat = target_model->GetMatrix();
70 int no = target_model->GetNodeIndex(object_node_name_);
71 if(no >= 0) {
72 mat = target_model->GetNodeMatrix(no);
73 }
74 }
75 }
76 mat = mul(rmat, mat);
77
78 return mat;
79}
