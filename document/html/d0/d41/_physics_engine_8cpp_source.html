96// Broad-phaseの各レイヤーはBroad-phaseの個別のバウンディングボリュームツリーになります。
97// 少なくとも、動かないオブジェクトと動くオブジェクトのレイヤーを用意して
98// フレームごとに静的オブジェクトでいっぱいのツリー更新を実行する必要がないようにする必要があります。
99//
100// オブジェクトレイヤーとBroad-phase
101// Layerとの間に1対1のマッピングが可能ですが、
102// 多くのオブジェクトレイヤーがある場合は多くのブロードフェーズツリーを作成することになり、効率的ではありません。
103// Broad-phase Layerを微調整したい場合 JPH_TRACK_BROADPHASE_STATS
104// を定義して出力メッセージで報告される統計情報を見てください。
105namespace BroadPhaseLayers {
106
107 constexpr JPH::BroadPhaseLayer NON_MOVING(0);
108 constexpr JPH::BroadPhaseLayer MOVING(1);
109 constexpr JPH::BroadPhaseLayer DEBRIS(2);
110 constexpr JPH::BroadPhaseLayer SENSOR(3);
111 constexpr JPH::uint NUM_LAYERS(4);
112
113 //! オブジェクトからBroad-phaseレイヤーへの変換テーブル
114 std::vector<JPH::BroadPhaseLayer> object_to_broad_phase_;
115
116 //! オブジェクトレイヤー同士が衝突するかどうかをカスタマイズするコールバック関数
117 std::function<bool(u16, u16)> can_object_layer_collide_;
118
119} // namespace BroadPhaseLayers

127class ObjectLayerPairFilter: public JPH::ObjectLayerPairFilter {
128 public:
129 virtual ~ObjectLayerPairFilter() = default;
130
131 //! 2つのオブジェクトレイヤーが衝突可能かどうかを判断するカスタム関数
132 //! @retval true 互いのレイヤーは衝突する
133 virtual bool ShouldCollide(JPH::ObjectLayer layer1, JPH::ObjectLayer layer2) const override {
134 // カスタム関数がある場合は実行
135 if(BroadPhaseLayers::can_object_layer_collide_) {
136 return BroadPhaseLayers::can_object_layer_collide_(layer1, layer2);
137 }
138
139 // デフォルトの設定
140 switch(layer1) {
141 case physics::ObjectLayers::NON_MOVING:
142 return layer2 == physics::ObjectLayers::MOVING || //
143 layer2 == physics::ObjectLayers::DEBRIS; //
144 case physics::ObjectLayers::MOVING:
145 return layer2 == physics::ObjectLayers::NON_MOVING || //
146 layer2 == physics::ObjectLayers::MOVING || //
147 layer2 == physics::ObjectLayers::SENSOR; //
148 case physics::ObjectLayers::DEBRIS:
149 return layer2 == physics::ObjectLayers::NON_MOVING;
150 case physics::ObjectLayers::SENSOR:
151 return layer2 == physics::ObjectLayers::MOVING;
152 default:
153 JPH_ASSERT(false);
154 return false;
155 }
156 }
157};

204//---------------------------------------------------------------------------
205//! Broad-phase層同士が衝突するかどうかを判定するカスタマイズ
206//---------------------------------------------------------------------------
207class ObjectVsBroadPhaseLayerFilter: public JPH::ObjectVsBroadPhaseLayerFilter {
208 public:
209 virtual ~ObjectVsBroadPhaseLayerFilter() = default;
210
211 //! @retval true 衝突あり
212 //! @retval true 衝突なし
213 virtual bool ShouldCollide(JPH::ObjectLayer layer1, JPH::BroadPhaseLayer layer2) const override {
214 auto layer1_broad_phase = BroadPhaseLayers::object_to_broad_phase_[layer1];
215
216 if(layer1_broad_phase == BroadPhaseLayers::NON_MOVING) {
217 return layer2 == BroadPhaseLayers::MOVING;
218 }
219 if(layer1_broad_phase == BroadPhaseLayers::MOVING) {
220 return layer2 == BroadPhaseLayers::NON_MOVING || //
221 layer2 == BroadPhaseLayers::MOVING || //
222 layer2 == BroadPhaseLayers::SENSOR; //
223 }
224 if(layer1_broad_phase == BroadPhaseLayers::DEBRIS) {
225 return layer2 == BroadPhaseLayers::NON_MOVING;
226 }
227 if(layer1_broad_phase == BroadPhaseLayers::SENSOR) {
228 return layer2 == BroadPhaseLayers::MOVING;
229 }
230
231 JPH_ASSERT(false);
232 return false;
233 }

407//---------------------------------------------------------------------------
408//! デフォルトコンストラクタ
409//---------------------------------------------------------------------------
410EngineImpl::EngineImpl() {
411 // デフォルトアロケーターを登録
412 JPH::RegisterDefaultAllocator();
413
414 jph_temp_allocator_ = std::make_unique<JPH::TempAllocatorImpl>(64 * 1024 * 1024);
415
416 physics_ = this;
417 temp_allocator_ = jph_temp_allocator_.get();
418
419 //----------------------------------------------------------
420 // トレースコールバックを登録
421 //----------------------------------------------------------
422 JPH::Trace = TraceImpl;
423 JPH_IF_ENABLE_ASSERTS(JPH::AssertFailed = AssertFailedImpl;)
424
425 //----------------------------------------------------------
426 // Factoryを作成
427 //----------------------------------------------------------
428 jph_factory_ = std::make_unique<JPH::Factory>();
429
430 // JoltPhysics内部変数に登録する
431 JPH::Factory::sInstance = jph_factory_.get();
432
433 //----------------------------------------------------------
434 // すべてのJoltPhysicsの型をFactoryに登録
435 //----------------------------------------------------------
436 JPH::RegisterTypes();
437
438 // 物理ジョブを複数スレッドで実行するジョブシステムが必要です。
439 // JoltPhysicsは基本的には自前のジョブスケジューラの上で実行することが出来ます。
440 // このJobSystemThreadPoolは実装例です。
441 job_system_ = std::make_unique<JPH::JobSystemThreadPool>(JPH::cMaxPhysicsJobs, JPH::cMaxPhysicsBarriers,
442 std::thread::hardware_concurrency() - 1);
443
444 //----------------------------------------------------------
445 // Physicsシステムを初期化
446 //----------------------------------------------------------
447 {
448 // Physicsシステムに追加できる剛体の最大量
449 // この個数以上追加しようとするとエラーが発生します。
450 constexpr JPH::uint MAX_BODIES = 65536;
451
452 // 剛体を同時アクセスから保護するために割り当てるべきミューテックスの数です。(default:0)
453 constexpr JPH::uint BODY_MUTEX_COUNT = 0;
454
455 // いつでもキューに入れられるボディペアの最大数
456 // Broad-phaseではバウンディングボックスに基づいて重複するボディペアを検出してNarrow-phaseのキューに挿入します
457 // このバッファを小さくしすぎると、キューが一杯になりBroad-phaseのジョブがNarrow-phaseの実行をし始めることになります。
458 // これは若干効率的ではありません。
459 constexpr JPH::uint MAX_BODY_PAIRS = 65536;
460
461 // コンタクト拘束バッファの最大サイズ
462 // この数よりも多くの接触（ボディ間の衝突）が検出された場合、これらの接触は無視されボディはワールド突き抜けて落下し始めます。
463 constexpr JPH::uint MAX_CONTACT_CONSTRAINTS = 65536;
464
465 // Physicsシステムを初期化作成
466 jph_physics_system_ = std::make_unique<JPH::PhysicsSystem>();
467
468 static ObjectVsBroadPhaseLayerFilter object_broad_phase_layer_filter_;
469 static ObjectLayerPairFilter object_layer_pair_filter_;
470
471 jph_physics_system_->Init(MAX_BODIES, BODY_MUTEX_COUNT, MAX_BODY_PAIRS, MAX_CONTACT_CONSTRAINTS,
472 broad_phase_layer_interface_, object_broad_phase_layer_filter_, object_layer_pair_filter_);
473
474 // パラメーター設定
475 JPH::PhysicsSettings settings{};
476
477 jph_physics_system_->SetPhysicsSettings(settings);
478
479 // 重力を設定
480 jph_physics_system_->SetGravity(JPH::Vec3(0.0f, -9.81f, 0.0f));
481
482 // ポインタを保存
483 physics_system_ = jph_physics_system_.get();
484 }
485
486 //----------------------------------------------------------
487 // ボディインターフェイスを保存
488 //----------------------------------------------------------
489 {
490 constexpr bool thread_safe = true;
491
492 if constexpr(thread_safe) {
493 body_interface_ = &jph_physics_system_->GetBodyInterface(); // Lockあり
494 } else {
495 body_interface_ = &jph_physics_system_->GetBodyInterfaceNoLock(); // Lockなし(利用には注意)
496 }
497 }
498
499 // ボディがアクティブになりスリープ状態になると、BodyActivationListenerが通知されます。
500 // これはジョブから呼び出されるので、ここで行うことはスレッドセーフである必要があることに注意してください。
501 // (登録は完全に任意です)
502 jph_physics_system_->SetBodyActivationListener(&body_activation_listener_);
503
504 // ボディが衝突する（しそうな）ときと再び離れるときに、ContactListenerに通知されます。
505 // これはジョブから呼び出されるので、ここで何をするにしてもスレッドセーフである必要があることに注意してください。
506 // (登録は完全に任意です)
507 jph_physics_system_->SetContactListener(&contact_listener_);
508
509 // デフォルトのBroadPhase変換テーブルを初期化
510 setLayerAlias(nullptr, 0);
511
512 is_valid_ = true;
513}

524//---------------------------------------------------------------------------
525//! 更新
526//---------------------------------------------------------------------------
527void EngineImpl::update(f32 dt) {
528 // 1/60秒より大きなステップを取る場合、シミュレーションを安定させるために
529 // 複数回の衝突ステップを行う必要があります。1/60秒ごとに1回の衝突ステップを実行します（切り上げ）
530 const u32 collision_steps = 1;
531
532 // より正確なステップ結果を得たい場合は、コリジョンステップの中で複数のサブステップを行うことができます。
533 const u32 integration_sub_steps = 1; // 通常は1に設定します。
534
535 // ワールドを時間経過させて更新
536 jph_physics_system_->Update(dt, collision_steps, integration_sub_steps, temp_allocator_, job_system_.get());
537}

539//---------------------------------------------------------------------------
540//! シーンにレイをキャスト
541//---------------------------------------------------------------------------
542bool EngineImpl::castRay(const Ray& ray, RayCastResult& result) {
543 JPH::RRayCast ray_cast{castJPH(ray.position_), castJPH(ray.dir_)};
544
545 auto& jph_narrow_phase = physics::Engine::physicsSystem()->GetNarrowPhaseQuery();
546
547 JPH::RayCastResult jph_result;
548 if(jph_narrow_phase.CastRay(ray_cast, //
549 jph_result, //
550 JPH::BroadPhaseLayerFilter{}, //
551 JPH::ObjectLayerFilter{}, //
552 JPH::BodyFilter{})) { //
553 result.t_ = jph_result.mFraction;
554 result.body_id_ = jph_result.mBodyID.GetIndexAndSequenceNumber();
555 return true;
556 }
557 return false;
558}

569//---------------------------------------------------------------------------
570//! シーン最適化を実行
571//---------------------------------------------------------------------------
572void EngineImpl::optimize() {
573 // 物理シミュレーションを開始する前にBroad-phaseを最適化することができます。
574 // これによって衝突検出のパフォーマンスが向上します
575 // これは高価な操作なので、毎フレームまたは新しいレベルセクションのストリーミング時などには絶対に呼び出さない方がよいでしょう。
576 // 代替策はすべての新しいオブジェクトを一度に
577 // 1つずつではなく、バッチで挿入してBroad-phaseを効率的に維持することです。
578 jph_physics_system_->OptimizeBroadPhase(); // ※このタイミングではまだオブジェクトが1つもないため無意味です。
579}

581//---------------------------------------------------------------------------
582//! オブジェクトレイヤーをカスタム設定
583//---------------------------------------------------------------------------
584void EngineImpl::setLayerAlias(const std::pair<physics::ObjectLayers, u16>* layers, size_t layerCount) {
585 std::array<JPH::BroadPhaseLayer, 4> table{
586 BroadPhaseLayers::NON_MOVING,
587 BroadPhaseLayers::MOVING,
588 BroadPhaseLayers::DEBRIS,
589 BroadPhaseLayers::SENSOR,
590 };
591
592 //----------------------------------------------------------
593 // オブジェクトからBroad-phaseレイヤーへの変換テーブル作成
594 //----------------------------------------------------------
595 if(layers) {
596 // 値の最大値を検索して変換テーブルを作成
597 // 都度検索すると負荷が高くなるため一発変換できる構造に変更しておく
598 auto* myLayerMax = std::max_element(layers, layers + layerCount, [](const auto& l, const auto& r) {
599 return l.second < r.second;
600 });
601
602 std::vector<physics::ObjectLayers> toObjectLayers_; // カスタム設定したレイヤー番号を内部番号に変換するテーブル
603
604 // 変換テーブルを最大数で設定
605 BroadPhaseLayers::object_to_broad_phase_.resize(static_cast<size_t>(myLayerMax->second) + 1);
606
607 for(size_t i = 0; i < layerCount; ++i) {
608 BroadPhaseLayers::object_to_broad_phase_[layers[i].second] = table[layers[i].first];
609 }
610 } else {
611 // デフォルト設定
612 BroadPhaseLayers::object_to_broad_phase_.resize(physics::ObjectLayers::MAX_COUNT);
613 BroadPhaseLayers::object_to_broad_phase_[physics::ObjectLayers::NON_MOVING] = BroadPhaseLayers::NON_MOVING;
614 BroadPhaseLayers::object_to_broad_phase_[physics::ObjectLayers::MOVING] = BroadPhaseLayers::MOVING;
615 BroadPhaseLayers::object_to_broad_phase_[physics::ObjectLayers::DEBRIS] = BroadPhaseLayers::DEBRIS;
616 BroadPhaseLayers::object_to_broad_phase_[physics::ObjectLayers::SENSOR] = BroadPhaseLayers::SENSOR;
617 }
618}
