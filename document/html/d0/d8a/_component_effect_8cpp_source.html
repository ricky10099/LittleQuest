16//! @brief Effekseerエフェクトロード
17//! @param path エフェクトファイル(.efkefc)
18void ComponentEffect::Load(std::string_view path) {
19 effect_status_.off(EffectBit::ErrorFileNotFound);
20
21 path_ = path;
22
23 // ファイルが存在しているか?
24 if(!HelperLib::File::CheckFileExistence(path_)) {
25 // ロードできなかった
26 effect_status_.on(EffectBit::ErrorFileNotFound);
27 effect_status_.off(EffectBit::Initialized);
28 return;
29 }
30
31 auto resource = exist_effects_resource_.find(path_);
32 if(resource != exist_effects_resource_.end()) {
33 // リソースが存在する場合
34 effect_handle_ = resource->second;
35
36 // 初期化済み
37 effect_status_.on(EffectBit::Initialized);
38 return;
39 }
40
41 int result = LoadEffekseerEffect(HelperLib::String::ToWString(path_).data());
42 if(result == -1) {
43 // ロードできなかった
44 effect_status_.on(EffectBit::ErrorFileNotFound);
45 return;
46 }
47
48 effect_handle_ = result;
49 // リソースの登録
50 exist_effects_resource_[path_] = effect_handle_;
51
52 // 初期化済み、未スタートにしておく
53 effect_status_.on(EffectBit::Initialized);
54}

63//! @brief モデル更新
64//! @param delta 1フレームの秒数
65void ComponentEffect::PostUpdate() {
66 float delta = GetDeltaTime();
67
68 // モデルが存在しているならばTransform設定を行う
69 if(IsValid() && IsPlaying()) {
70 // アニメーションがあり再生している?
71 effect_time_ += (delta * effect_speed_);
72
73 auto mat = effect_transform_;
74 auto trns = GetOwner()->GetComponent<ComponentTransform>();
75 if(trns)
76 mat = mul(mat, trns->GetMatrix());
77
78 // Drawはここで抑えておく
79 if(GetStatus(Component::StatusBit::NoDraw)) {
80 //システムして「すべて描くモード」が使用されているため消すことができない。
81 //よってスケールを0にして表示するようにして消したように見せかける
82 SetScalePlayingEffekseer3DEffect(effect_play_handle_, 0, 0, 0);
83 return;
84 }
85
86 float* matz = (float*)mat.f32_128_0;
87 float pos[3], rot[3], scale[3];
88 DecomposeMatrixToComponents(matz, pos, rot, scale);
89
90 SetPosPlayingEffekseer3DEffect(effect_play_handle_, pos[0], pos[1], pos[2]);
91 SetScalePlayingEffekseer3DEffect(effect_play_handle_, scale[0], scale[1], scale[2]);
92 SetRotationPlayingEffekseer3DEffect(effect_play_handle_, radians((float1)rot[0]), radians((float1)rot[1]),
93 radians((float1)rot[2]));
94 }
95}

97//! @brief モデル描画
98void ComponentEffect::Draw() {
99 if(!effect_status_.is(EffectBit::Initialized))
100 return;
101
102 if(Scene::IsPause() || effect_status_.is(EffectBit::Paused)) {
103 SetSpeedPlayingEffekseer3DEffect(effect_play_handle_, 0.0f);
104 } else {
105 SetSpeedPlayingEffekseer3DEffect(effect_play_handle_, effect_speed_);
106 }
107
108 bool playing = IsEffekseer3DEffectPlaying(effect_play_handle_) == 0 ? !effect_status_.is(EffectBit::Paused) : false;
109 effect_status_.set(EffectBit::Playing, playing);
110 if(!playing && effect_status_.is(EffectBit::Loop)) {
111 Play(true);
112 }
113}

129 ImGui::Begin(obj_name.data());

138

142 ImGui::BeginDisabled(true); // UI上の編集不可(ReadOnly)
143 {
144 if(loaded)
145 ImGui::Checkbox(u8"【LoadOK】", &loaded);
146 else
147 ImGui::TextColored({1, 0, 0, 1}, u8"【LoadNG】");
148 }
149 ImGui::EndDisabled();
150
151 ImGui::SameLine();
152
153 // ファイル名
154 char file_name[1024];
155 sprintf_s(file_name, "%s", path_.c_str());
156 if(ImGui::InputText(u8"File", file_name, 1024)) {
157 path_ = file_name;
158 effect_status_.off(EffectBit::Initialized);
159 Load(path_);
160 }
161 ImGui::Separator();
162
163 if(IsValid()) {
164 ImGui::CheckboxFlags("Loop", &effect_status_.get(), 1 << (int)EffectBit::Loop);
165
166 if(IsPaused()) {
167 if(ImGui::Button("Resume")) {
168 PlayPause(false);
169 }
170 } else {
171 if(!IsPlaying()) {
172 if(ImGui::Button("Play")) {
173 Play();
174 }
175 } else {
176 if(ImGui::Button("Pause")) {
177 PlayPause();
178 }
179 }
180 }
181 ImGui::Separator();
182 }
183
184 // アニメーション名
185 if(IsPlaying()) {
186 ImGui::TextColored({0.5, 1, 0.5, 1}, u8"再生中");
187 ImGui::Text(u8"[%3.2f]%s", effect_time_, GetEffectName().data());
188 ImGui::Separator();
189 }
190
191 // モデル姿勢
192 if(ImGui::TreeNode(u8"エフェクト姿勢")) {
193 ImGui::DragFloat4(u8"Ｘ軸", effect_transform_.f32_128_0, 0.01f, -10000.0f, 10000.0f, "%.2f");
194 ImGui::DragFloat4(u8"Ｙ軸", effect_transform_.f32_128_1, 0.01f, -10000.0f, 10000.0f, "%.2f");
195 ImGui::DragFloat4(u8"Ｚ軸", effect_transform_.f32_128_2, 0.01f, -10000.0f, 10000.0f, "%.2f");
196 ImGui::DragFloat4(u8"座標", effect_transform_.f32_128_3, 0.01f, -10000.0f, 10000.0f, "%.2f");
197 ImGui::Separator();
198 ImGui::TreePop();
199 }
200
201 // 姿勢を TRSで変更できるように設定
202 float* mat = effect_transform_.f32_128_0;
203 float matrixTranslation[3], matrixRotation[3], matrixScale[3];
204 DecomposeMatrixToComponents(mat, matrixTranslation, matrixRotation, matrixScale);
205 ImGui::DragFloat3(u8"座標(T)", matrixTranslation, 0.01f, -100000.00f, 100000.0f, "%.2f");
206 ImGui::DragFloat3(u8"回転(R)", matrixRotation, 0.1f, -360.0f, 360.0f, "%.2f");
207 ImGui::DragFloat3(u8"サイズ(S)", matrixScale, 0.01f, 0.00f, 1000.0f, "%.2f");
208 RecomposeMatrixFromComponents(matrixTranslation, matrixRotation, matrixScale, mat);
209
210 ImGui::TreePop();
211 }
212 }
213 ImGui::End();
214}
