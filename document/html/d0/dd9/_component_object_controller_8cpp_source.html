10void ComponentObjectController::Update() {
11 __super::Update();
12
13 // オーナー(自分がAddComponentされたObject)を取得します
14 // 処理されるときは必ずOwnerは存在しますので基本的にnullptrチェックは必要ありません
15 auto owner = GetOwner();
16
17 // 移動方向
18 float3 dir{0, 0, 0};
19 if(IsKey(key_up_))
20 dir += {0, 0, -1};
21
22 if(IsKey(key_down_))
23 dir += {0, 0, 1};
24
25 if(IsKey(key_right_))
26 dir += {-1, 0, 0};
27
28 if(IsKey(key_left_))
29 dir += {1, 0, 0};
30
31 // 移動キーが押されているか?
32 if((float)length(dir) > 0.0f) {
33 // 斜めが押されていることを考慮し、
34 // その方向の移動スピードを1とし、スピードを掛け合わせる
35 dir = normalize(dir);
36
37 // キャラのローカル方向で移動をさせる
38 owner->AddTranslate(dir * move_speed_, true);
39
40 // モデルを移動の方向に向けます
41 if(auto mdl = owner->GetComponent<ComponentModel>()) {
42 mdl->SetRotationToVectorWithLimit(dir, rot_speed_);
43 mdl->PlayAnimationNoSame("walk", true);
44 }
45 } else {
46 // モデルを移動の方向に向けます
47 if(auto mdl = owner->GetComponent<ComponentModel>())
48 mdl->PlayAnimationNoSame("idle", true);
49 }
50
51 // カメラが存在している場合
52 if(auto camera = Scene::GetCurrentCamera().lock()) {
53 // SpringArmのオブジェクトが自分の場合のみ
54 auto cam_owner = camera->GetOwner();
55 if(auto arm = cam_owner->GetComponent<ComponentSpringArm>()) {
56 // そのターゲットはオーナーのオブジェクトであることが条件となる
57 auto obj = arm->GetSpringArmObject().lock();
58 if(obj.get() == owner) {
59 // オブジェクトコントローラーにターゲットがいる場合
60 if(auto target = target_.lock()) {
61 // 徐々に敵の方に向ける(1フレーム最大3度)
62 owner->SetRotationToPositionWithLimit(target->GetTranslate(), target_cam_side_speed_ * GetDeltaTime60());
63
64 // カメラローテーションをロック方向にしておくと
65 // 戻った時に違和感がない
66 cam_ry_ = owner->GetRotationAxisXYZ().y;
67 cam_rx_ = (cam_rx_ + target_cam_up_down) * 0.95f - target_cam_up_down;
68
69 arm->SetSpringArmRotate({cam_rx_, 0, 0});
70 } else {
71 // ターゲットがない場合
72 // マウスで動作せる場合の処理
73 float mouse_ud = mouse_up_down_ * GetDeltaTime60();
74 float mouse_lr = mouse_left_right_ * GetDeltaTime60();
75 float cam_spd = cam_speed_ * GetDeltaTime60();
76 if(use_mouse_) {
77 cam_rx_ += GetMouseMoveY() * (mouse_ud / 100.0f);
78 if(cam_rx_ > limit_cam_up_)
79 cam_rx_ = limit_cam_up_;
80 if(cam_rx_ < limit_cam_down_)
81 cam_rx_ = limit_cam_down_;
82 cam_ry_ += GetMouseMoveX() * (mouse_lr / 100.0f);
83 if(cam_ry_ > 360.0f)
84 cam_ry_ -= 360.0f;
85 if(cam_ry_ < -360.0f)
86 cam_ry_ += 360.0f;
87 }
88 // キーでカメラを移動させる
89 if(IsKey(cam_up_)) {
90 cam_rx_ += cam_spd;
91 if(cam_rx_ > limit_cam_up_)
92 cam_rx_ = limit_cam_up_;
93 }
94 if(IsKey(cam_down_)) {
95 cam_rx_ -= cam_spd;
96 if(cam_rx_ < limit_cam_down_)
97 cam_rx_ = limit_cam_down_;
98 }
99 if(IsKey(cam_right_)) {
100 cam_ry_ += cam_spd;
101 if(cam_ry_ > 360.0f)
102 cam_ry_ -= 360.0f;
103 }
104 if(IsKey(cam_left_)) {
105 cam_ry_ -= cam_spd;
106 if(cam_ry_ < -360.0f)
107 cam_ry_ += 360.0f;
108 }
109
110 // Y軸はオブジェクトそのものの向きを使う
111 owner->SetRotationAxisXYZ({0, cam_ry_, 0});
112 // 上下はSpringArmで制御しておく
113 arm->SetSpringArmRotate({cam_rx_, 0, 0});
114 }
115 }
116 }
117 }
118}
