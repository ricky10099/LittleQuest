259 //! @brief ベクトルの方向へ向ける
260 //! @param vec 向きたいベクトル
261 //! @param limit 向ける限界角度
262 //! @param up_change Y軸も変更するか?
263 //! @return 自分のSharedPtr
264 auto SetRotationToVectorWithLimit(float3 vec, float limit, float3 up = {0, 1, 0}, bool up_change = false) {
265 float delta = GetDeltaTime60();
266 limit = limit * delta;
267
268 // 今の姿勢マトリクス
269 auto now_mat = GetMatrix();
270
271 // 向きたい方向のマトリクス
272 auto mat = HelperLib::Math::CreateMatrixByFrontVector(-vec, up, up_change);
273
274 // 現在の姿勢
275 float matrixTranslation[3], matrixRotation[3], matrixScale[3];
276 // 向きたい姿勢
277 float matrixTranslationVec[3], matrixRotationVec[3], matrixScaleVec[3];
278
279 DecomposeMatrixToComponents(GetMatrixFloat(), matrixTranslation, matrixRotation, matrixScale);
280
281 DecomposeMatrixToComponents(mat.f32_128_0, matrixTranslationVec, matrixRotationVec, matrixScaleVec);
282
283 float rot_x = matrixRotationVec[0] - matrixRotation[0];
284 if(rot_x > 180)
285 rot_x -= 360;
286 if(rot_x < -180)
287 rot_x += 360;
288 float rot_y = matrixRotationVec[1] - matrixRotation[1];
289 if(rot_y > 180)
290 rot_y -= 360;
291 if(rot_y < -180)
292 rot_y += 360;
293 float rot_z = matrixRotationVec[2] - matrixRotation[2];
294 if(rot_z > 180)
295 rot_z -= 360;
296 if(rot_z < -180)
297 rot_z += 360;
298
299 if(rot_x > 0 && rot_x > limit)
300 rot_x = limit;
301 if(rot_x < 0 && rot_x < -limit)
302 rot_x = -limit;
303
304 if(rot_y > 0 && rot_y > limit)
305 rot_y = limit;
306 if(rot_y < 0 && rot_y < -limit)
307 rot_y = -limit;
308
309 if(rot_z > 0 && rot_z > limit)
310 rot_z = limit;
311 if(rot_z < 0 && rot_z < -limit)
312 rot_z = -limit;
313
314 matrixRotationVec[0] = rot_x + matrixRotation[0];
315 matrixRotationVec[1] = rot_y + matrixRotation[1];
316 matrixRotationVec[2] = rot_z + matrixRotation[2];
317 RecomposeMatrixFromComponents(matrixTranslation, matrixRotationVec, matrixScale, GetMatrixFloat());
318 return SharedThis();
319 }
