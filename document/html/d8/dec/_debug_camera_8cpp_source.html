37 void PreDraw() override {
38 auto cam = GetComponent<ComponentCamera>();
39
40 auto LR = (GetMouseInput() & (MOUSE_INPUT_LEFT | MOUSE_INPUT_RIGHT)) == (MOUSE_INPUT_LEFT | MOUSE_INPUT_RIGHT);
41 auto M = (GetMouseInput() & MOUSE_INPUT_MIDDLE);
42
43 [[maybe_unused]] auto L = (GetMouseInput() & MOUSE_INPUT_LEFT);
44 auto R = (GetMouseInput() & MOUSE_INPUT_RIGHT);
45
46 float2 mouse_now = float2{GetMouseX(), GetMouseY()};
47 mouse_vec_ = mouse_now - mouse_xy_;
48 mouse_xy_ = mouse_now;
49
50 if(ImGui::IsAnyItemActive())
51 return;
52 if(ImGuizmo::IsUsing())
53 return;
54
55 // UnrealEngineのみ仮作成
56 if(control == UnrealEngine) {
57 // カメラに合わせた軸の作成
58 float3 back = normalize(cam_pos_ - tgt_pos_);
59 float3 up = {0, 1, 0};
60 float3 right = cross(up, back);
61 float3 front = -back;
62
63 // wheel
64 wheel_val = GetMouseWheelRotVolF();
65
66 if(wheel_val != 0) {
67 front *= wheel_val * 10;
68 cam_pos_ = cam_pos_ + front;
69 tgt_pos_ = tgt_pos_ + front;
70 }
71
72 // 平行移動
73 if(M || LR) {
74 float3 mv = (right * -mouse_vec_.x + up * -mouse_vec_.y) * 0.1f;
75
76 cam_pos_ = cam_pos_ + mv;
77 tgt_pos_ = tgt_pos_ + mv;
78 } else if(R) {
79 auto tgt_new_ = tgt_pos_ - cam_pos_;
80
81 auto rot_y = matrix::rotateY(mouse_vec_.x * 0.003f);
82 auto rot_x = matrix::rotateAxis(right, -mouse_vec_.y * 0.003f);
83
84 tgt_pos_ = mul(float4(tgt_new_, 1), mul(rot_y, rot_x)).xyz + cam_pos_;
85
86 if(Input::IsKey(KEY_INPUT_W)) {
87 tgt_pos_ += front * 2.0f;
88 cam_pos_ += front * 2.0f;
89 }
90 if(Input::IsKey(KEY_INPUT_A)) {
91 tgt_pos_ += right * 2.0f;
92 cam_pos_ += right * 2.0f;
93 }
94 if(Input::IsKey(KEY_INPUT_D)) {
95 tgt_pos_ -= right * 2.0f;
96 cam_pos_ -= right * 2.0f;
97 }
98 if(Input::IsKey(KEY_INPUT_S)) {
99 tgt_pos_ += back * 2.0f;
100 cam_pos_ += back * 2.0f;
101 }
102 if(Input::IsKey(KEY_INPUT_Q)) {
103 tgt_pos_ += up * 2.0f;
104 cam_pos_ += up * 2.0f;
105 }
106 if(Input::IsKey(KEY_INPUT_E)) {
107 tgt_pos_ -= up * 2.0f;
108 cam_pos_ -= up * 2.0f;
109 }
110 } else if(L) {
111 auto tgt_new_ = tgt_pos_ - cam_pos_;
112
113 auto rot_y = matrix::rotateY(mouse_vec_.x * 0.003f);
114 // auto rot_x = matrix::rotateAxis( right, -mouse_vec_.y
115 // * 0.003f );
116
117 auto f_v = front;
118 f_v.y = 0.0f; // 地面に沿った平行な向きに進む
119
120 tgt_pos_ = mul(float4(tgt_new_, 1), rot_y).xyz + cam_pos_ + f_v * (-mouse_vec_.y);
121 cam_pos_ += f_v * (-mouse_vec_.y);
122 }
123 // フォーカス
124 if(Input::IsKeyDown(KEY_INPUT_F)) {
125 auto f_v = normalize(tgt_pos_ - cam_pos_);
126 if(auto obj = Scene::SelectObjectPtr()) {
127 tgt_pos_ = obj->GetTranslate();
128 cam_pos_ = obj->GetTranslate() - f_v * 30.0f;
129 }
130 }
131
132 mat_view_ = matrix::lookAtLH(cam_pos_, tgt_pos_);
133
134 cam->SetPositionAndTarget(cam_pos_, tgt_pos_);
135 cam->SetCameraStatus(ComponentCamera::CameraBit::DebugCamera, true);
136
137 SetCameraViewMatrix(mat_view_); // ビュー行列
138 SetupCamera_ProjectionMatrix(mat_proj_); // 投影行列
139 }
140 }
