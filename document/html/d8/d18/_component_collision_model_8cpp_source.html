119void ComponentCollisionModel::AttachToModel(bool update) {
120 if(auto mdl = GetOwner()->GetComponent<ComponentModel>()) {
121#ifdef USE_JOLT_PHYSICS
122 // 地形衝突情報を作成 (メッシュ剛体は必ず静的)

124 bool same = (fabs(scale.x - scale.y) < FLT_EPSILON) && (fabs(scale.x - scale.z) < FLT_EPSILON);

126 assert("マップのスケールのXYZが異なります。[無視する]で平均をとります." && same);

128 RigidBody() = physics::createRigidBody(shape::Mesh(mdl->GetModelClass().get(), s), // メッシュ

199//! @brief マップに対する正確な移動量を割り出す(傾きによる移動量)
200//! @param vec 動きたい移動量
201//! @param force 山に上る時の上りにくさ (1.0で普通に楽々上る
202//! ~ 3.0くらいだともう登れない )
203//! @return 実際動ける量
204float3 ComponentCollisionModel::checkMovement(float3 pos, float3 vec, float force) {
205 MV1_COLL_RESULT_POLY hit_poly{};
206 float3 top = pos + float3(0, 10, 0);
207 float3 btm = pos + float3(0, -1.5, 0);
208
209 if(length(vec).x <= 0)
210 return float3(0, 0, 0);
211
212 if(auto mdl = GetOwner()->GetComponent<ComponentModel>()) {
213 hit_poly = MV1CollCheck_Line(mdl->GetModel(), -1, cast(top), cast(btm));
214 if(hit_poly.HitFlag) {
215 // 制限あり (dotがマイナスということは山に上る形になっている)
216 float pt = dot(cast(hit_poly.Normal), normalize(vec));
217 if(pt < 0) {
218 pt *= force;
219 if(pt < -1)
220 pt = -1;
221
222 // 傾きに合わせて、実際の移動量は、vec(最大) ~ 0(最小) となる
223 return vec + vec * pt;
224 }
225 }
226 }
227 return vec;
228}
