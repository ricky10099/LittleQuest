58//! @brief 当たった情報はコールバックで送られてくる
59//! @param hitInfo 当たった情報
60//! @details 当たった回数分ここに来ます
61void ComponentCollision::OnHit(const HitInfo& hitInfo) {
62 auto obj = GetOwner();
63
64 // Staticな物質にぶつかった場合、gravity_を下げる
65 if(hitInfo.hit_collision_->GetMass() < 0) {
66 auto vec = obj->GetTranslate() - obj->GetOldWorldMatrix().translate();
67 float3 nvc = {0.0f, -1.0f, 0.0f};
68 if(length(vec).x <= 0 || length(now_gravity_).x <= 0) {
69 now_gravity_ = 0.0f;
70 } else {
71 nvc = normalize(vec);
72 float d = dot(normalize(now_gravity_), nvc);
73 now_gravity_ *= (1 - d * d * 0.1f);
74 }
75 }
76
77 obj->OnHit(hitInfo);
78}

253//! @brief Capsule VS Sphere
254//! @param col1 Capsuleコリジョン
255//! @param col2 Sphere コリジョン
256//! @return 当たり情報
257ComponentCollision::HitInfo ComponentCollision::isHit(ComponentCollisionCapsulePtr col1, ComponentCollisionSpherePtr col2) {
258 ComponentCollision::HitInfo info{};
259
260 // 自分のコリジョン
261 float3 cpos1 = col1->GetTranslate();
262 float3 cpos2 = col1->GetVectorAxisY() * col1->GetHeight() + cpos1;
263 float cs = 1.0f; //< スケール
264
265 float3 epos1;
266 float es = 1.0f;
267
268 // モデルアタッチ
269 if(col1->attach_node_ >= 0) {
270 if(auto mdl = col1->GetOwner()->GetComponent<ComponentModel>()) {
271 cpos1 = mul(float4(cpos1, 1), col1->attach_node_matrix_).xyz;
272 cpos2 = mul(float4(cpos2, 1), col1->attach_node_matrix_).xyz;
273 cpos2 = normalize(cpos2 - cpos1) * col1->GetHeight() + cpos1;
274 }
275 } else {
276 // ComponentTransform(オブジェクト姿勢)
277 if(auto cmp = col1->GetOwner()->GetComponent<ComponentTransform>()) {
278 // 高さに回転とスケールを掛け合わせる
279 cpos1 = mul(float4(cpos1, 1), cmp->GetMatrix()).xyz;
280 cpos2 = mul(float4(cpos2, 1), cmp->GetMatrix()).xyz;
281 // 半径はXZで平均としておく
282 cs = (length(cmp->GetMatrix().axisX()) + length(cmp->GetMatrix().axisZ())) / 2;
283 }
284 }
285
286 // モデルアタッチ
287 if(col2->attach_node_ >= 0) {
288 if(auto mdl = col1->GetOwner()->GetComponent<ComponentModel>()) {
289 epos1 = col2->GetTranslate();
290 epos1 = mul(float4(epos1, 1), col2->attach_node_matrix_).xyz;
291 }
292 } else {
293 // ComponentTransform(オブジェクト姿勢)
294 if(auto cmp = col2->GetOwner()->GetComponent<ComponentTransform>()) {
295 epos1 = mul(col2->GetMatrix(), cmp->GetMatrix())._41_42_43;
296 // pos1 = mul( float4( pos1, 0 ) , cmp->GetMatrix() ).xyz;
297 // pos1 += cmp->GetTranslate().xyz;
298 float sx = length(cmp->GetVectorAxisX());
299 float sy = length(cmp->GetVectorAxisY());
300 float sz = length(cmp->GetVectorAxisZ());
301 es = (sx + sy + sz) / 3.0f;
302 }
303 }
304
305 float cr = col1->GetRadius() * cs;
306 float3 cv = normalize(cpos1 - cpos2);
307 VECTOR c1 = cast(cpos1 - cv * cr);
308 VECTOR c2 = cast(cpos2 + cv * cr);
309
310 float er = col2->GetRadius() * es;
311 VECTOR e1 = cast(epos1);
312
313 // 跳ね返り点が欲しいため、HitCheck_Capsule_Capsuleは使わない
314 SEGMENT_POINT_RESULT result;
315 Segment_Point_Analyse(&c1, &c2, &e1, &result);
316
317 if(result.Seg_Point_MinDist_Square < (cr + er) * (cr + er)) {
318 // 線と線で一番近くなる点を求め、ベクトル化する
319 // 最も近い点
320 float3 c0 = cast(result.Seg_MinDist_Pos);
321 float3 e0 = cast(e1);
322 float3 vec = e0 - c0; // 調べたほうの跳ね返りの方向(100%)
323 float len = length(vec);
324 if(abs(len) <= abs(len) * FLT_EPSILON) {
325 // 全く同じ位置にいる場合はz移動する形にしておく
326 vec = {0, 0, 1};
327 }
328
329 float3 vs = normalize(vec) * (cr + er);
330 vec -= vs;
331
332 // このpush_は、調べたほうの押し戻し方向100%で作成する
333 info.push_ = vec;
334 info.hit_ = true;
335 info.hit_position_ = (e0 + c0) * 0.5f;
336 }
337
338 return info;
339}

383 float3 epos1 = col2->GetTranslate();

392 epos2 = normalize(epos2 - epos1) * col2->GetHeight() + epos1;
393 }
394 } else {
395 // ComponentTransform(オブジェクト姿勢)
396 if(auto cmp = col2->GetOwner()->GetComponent<ComponentTransform>()) {

398 // 高さに回転とスケールを掛け合わせる
399 epos1 = mul(float4(epos1, 1), mtx).xyz;
400 epos2 = mul(float4(epos2, 1), mtx).xyz;
401 // 半径はXZで平均としておく
402 es = (length(mtx.axisX()) + length(mtx.axisZ())) / 2;
403 }
404 }
405
406 float cr = col1->GetRadius() * cs;
407 float3 cv = normalize(cpos1 - cpos2);
408 VECTOR c1 = cast(cpos1 - (cv * cr));

427 if(result.SegA_SegB_MinDist_Square < (cr + er) * (cr + er)) {
428 // 線と線で一番近くなる点を求め、ベクトル化する

436 // 全く同じ位置にいる場合はz移動する形にしておく
437 vec = {0, 0, 1};

444 info.hit_ = true;
445 info.hit_position_ = (e0 + c0) * 0.5f;
446 }
447
448 return info;
449}

455ComponentCollision::HitInfo ComponentCollision::isHit(ComponentCollisionSpherePtr col1, ComponentCollisionSpherePtr col2) {
456 ComponentCollision::HitInfo info{};
457
458 float3 pos1;
459 float scale1 = 1.0f;
460
461 // モデルアタッチ
462 if(col1->attach_node_ >= 0) {
463 if(auto mdl = col1->GetOwner()->GetComponent<ComponentModel>()) {
464 pos1 = col1->GetTranslate();
465 pos1 = mul(float4(pos1, 1), col1->attach_node_matrix_).xyz;
466 }
467 } else {
468 if(auto cmp = col1->GetOwner()->GetComponent<ComponentTransform>()) {
469 pos1 = mul(col1->GetMatrix(), cmp->GetWorldMatrix())._41_42_43;
470 // pos1 = mul( float4( pos1, 0 ) , cmp->GetMatrix() ).xyz;
471 // pos1 += cmp->GetTranslate().xyz;
472 float sx = length(cmp->GetVectorAxisX());
473 float sy = length(cmp->GetVectorAxisY());
474 float sz = length(cmp->GetVectorAxisZ());
475 scale1 = (sx + sy + sz) / 3.0f;
476 }
477 }
478
479 float3 pos2;
480 float scale2 = 1.0f;
481
482 // モデルアタッチ
483 if(col2->attach_node_ >= 0) {
484 if(auto mdl = col2->GetOwner()->GetComponent<ComponentModel>()) {
485 pos2 = col2->GetTranslate();
486 pos2 = mul(float4(pos2, 1), col2->attach_node_matrix_).xyz;
487 }
488 } else {
489 if(auto cmp = col2->GetOwner()->GetComponent<ComponentTransform>()) {
490 pos2 = mul(col2->GetMatrix(), cmp->GetWorldMatrix())._41_42_43;
491 // pos2 = mul( float4( pos2, 0 ), cmp->GetMatrix() ).xyz;
492 // pos2 += cmp->GetTranslate().xyz;
493 float sx = length(cmp->GetVectorAxisX());
494 float sy = length(cmp->GetVectorAxisY());
495 float sz = length(cmp->GetVectorAxisZ());
496 scale2 = (sx + sy + sz) / 3.0f;
497 }
498 }
499
500 if(HitCheck_Sphere_Sphere(cast(pos1), col1->GetRadius() * scale1, cast(pos2), col2->GetRadius() * scale2)) {
501 // 中間地点を当たった場所にする
502 info.hit_ = true;
503 info.hit_position_ = (pos1 - pos2) * 0.5f + pos2;
504
505 // 押し出し方向
506 float3 distance = pos1 - pos2;
507 float len = length(distance);
508 if(abs(len) <= abs(len) * FLT_EPSILON) {
509 // 全く同じ位置にいる場合はz移動する形にしておく
510 distance = {0, 0, 1};
511 }
512
513 float3 vec = normalize(distance);
514 float reallen = col1->GetRadius() * scale1 + col2->GetRadius() * scale2;
515 vec = vec * (reallen - len);
516
517 // このpush_は、調べたほうの押し戻し方向100%で作成する
518 info.push_ = vec;
519 }
520
521 return info;
522}

524//! @brief Model VS Sphere
525//! @param col1 Model コリジョン
526//! @param col2 Sphereコリジョン
527//! @return 当たり情報
528ComponentCollision::HitInfo ComponentCollision::isHit(ComponentCollisionSpherePtr col1, ComponentCollisionModelPtr col2) {
529 ComponentCollision::HitInfo info{};
530
531 // モデルが存在していない
532 auto mdl = col2->GetOwner()->GetComponent<ComponentModel>();
533 if(mdl == nullptr)
534 return info;
535
536 float3 opos{};
537 float3 cpos{};
538 if(col1->attach_node_ >= 0) {
539 cpos = mul(float4(col1->GetTranslate(), 1), col1->attach_node_matrix_).xyz;
540 opos = cpos;
541 } else {
542 // オブジェクト位置に対するコリジョン(1フレーム前)
543 opos = mul(float4(col1->GetTranslate(), 1), col1->GetOwner()->GetOldWorldMatrix()).xyz;
544
545 // オブジェクト位置に対するコリジョン
546 cpos = mul(float4(col1->GetTranslate(), 1), col1->GetOwner()->GetWorldMatrix()).xyz;
547
548 // 実際の移動できる量にする
549 auto move = cpos - opos;
550 float movey = move.y;
551 move.y = 0;
552
553 if(col1->IsUseGravity()) {
554 move = col2->checkMovement(opos, move, difficult_to_climb_);
555 move.y = movey;
556 cpos = opos + move;
557
558 // 当たりからキャラの位置を求める
559 auto rot = col1->GetOwner()->GetWorldMatrix();
560 rot._41_42_43 = float3{0.0f, 0.0f, 0.0f};
561 auto col1r = mul(float4(col1->GetTranslate(), 1), rot).xyz;
562 col1->GetOwner()->Translate() = cpos - col1r;
563 }
564 }
565
566 float scale = 1.0f;
567 if(col1->attach_node_ < 0) {

982 VECTOR center = cast(pos);
983
