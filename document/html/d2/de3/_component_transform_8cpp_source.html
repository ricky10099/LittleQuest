21bool ShowGizmo(float* matrix, ImGuizmo::OPERATION ope, ImGuizmo::MODE mode, uint64_t id) {
22 // Gizmoを表示するためのMatrixをDxLibから取得
23 auto camera_view = GetCameraViewMatrix();
24 auto camera_proj = GetCameraProjectionMatrix();

30 // 画面サイズを取得する

54}

56//! @brief Matrixから位置・回転・スケールに変換(ZXYで変換)
57//! @param matx 入力Matrix
58//! @param translation 位置
59//! @param rotation 回転
60//! @param scale スケール
61void DecomposeMatrixToComponents(const float* matx, float* translation, float* rotation, float* scale) {
62 matrix mat = *(matrix*)matx;
63
64 scale[0] = length(mat.axisX());
65 scale[1] = length(mat.axisY());
66 scale[2] = length(mat.axisZ());
67
68 MATRIX dxmat = cast(mat);
69
70 // ZXYで回転を作成 (DxLib命令を利用)
71 GetMatrixZXYRotation(&dxmat, &rotation[0], &rotation[1], &rotation[2]);
72
73 constexpr float RAD2DEG = (180.f / DX_PI_F);
74
75 rotation[0] *= RAD2DEG;
76 rotation[1] *= RAD2DEG;
77 rotation[2] *= RAD2DEG;
78
79 translation[0] = mat.translate().x;
80 translation[1] = mat.translate().y;
81 translation[2] = mat.translate().z;
82}

84//! @brief 位置・回転・スケールからMatrixに変換(ZXYで変換)
85//! @param translation 位置
86//! @param rotation 回転
87//! @param scale スケール
88//! @param matx マトリクス
89void RecomposeMatrixFromComponents(const float* translation, const float* rotation, const float* scale, float* matx) {
90 constexpr float DEG2RAD = (DX_PI_F / 180.0f);
91 MATRIX dxmat;
92
93 // ZXYでマトリクスを作成 (DxLib命令を利用)
94 CreateRotationZXYMatrix(&dxmat, rotation[0] * DEG2RAD, rotation[1] * DEG2RAD, rotation[2] * DEG2RAD);
95
96 matrix mat = cast(dxmat);
97
98 mat._11_12_13 = mat._11_12_13 * scale[0];
99 mat._21_22_23 = mat._21_22_23 * scale[1];
100 mat._31_32_33 = mat._31_32_33 * scale[2];
101 mat._41_42_43_44 = {translation[0], translation[1], translation[2], 1};
102
103 memcpy(matx, &mat, sizeof(mat));
104}
