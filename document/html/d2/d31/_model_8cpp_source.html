21//! 読み込み
22//---------------------------------------------------------------------------
23bool Model::load(std::string_view path) {
24 if(!textureIBL_diffuse_)
25 textureIBL_diffuse_ = std::make_shared<Texture>("data/IBL/iblDiffuseHDR.dds");
26 if(!textureIBL_specular_)
27 textureIBL_specular_ = std::make_shared<Texture>("data/IBL/iblSpecularHDR.dds");
28 //----------------------------------------------------------
29 // モデルリソース読み込み
30 //----------------------------------------------------------
31 {
32 std::string resource_path = std::string(path);
33
34 auto it = resource_model_pool.find(resource_path);
35 if(it == resource_model_pool.end()) { // 新規登録
36 resource_model_pool[resource_path] = std::make_shared<ResourceModel>(path);
37 }
38
39 // 共有
40 resource_model_ = resource_model_pool[resource_path];
41 }
42
43 // 以前ロードしたデータは削除する
44 if(animation_) {
45 animation_->bindModel(nullptr);
46 }
47
48 // [DxLib] MV1モデルを解放 (複製されたハンドル)
49 if(mv1_handle_ != -1) {
50 MV1DeleteModel(mv1_handle_);
51 mv1_handle_ = -1;
52 }
53
54 // 初期化を必要とするようにする
55 need_initialize_ = true;
56
57 // 初回のみ読み込み
58 if(shader_vs_ == nullptr || shader_ps_ == nullptr) {
59 //----------------------------------------------------------
60 // シェーダーを読み込み
61 //----------------------------------------------------------
62 // 頂点シェーダー
63 // DX_MV1_VERTEX_TYPE_NUM 個のシェーダーバリエーションを生成
64 shader_vs_ = std::make_shared<ShaderVs>("data/Shader/vs_model", VS_VARIANT_COUNT);
65
66 // ピクセルシェーダー
67 shader_ps_ = std::make_shared<ShaderPs>("data/Shader/ps_model", PS_VARIANT_COUNT);
68
69 //----------------------------------------------------------
70 // デフォルトテクスチャを読み込み
71 //----------------------------------------------------------
72 tex_null_white_ = std::make_shared<Texture>("data/System/null_white.dds");
73 tex_null_black_ = std::make_shared<Texture>("data/System/null_black.dds");
74 tex_null_normal_ = std::make_shared<Texture>("data/System/null_normal.dds");
75 }
76
77 return resource_model_->isValid();
78}

86//! 描画

88void Model::render(ShaderVs* override_vs, ShaderPs* override_ps) {
89 if(!resource_model_)
90 return;
91
92 if(!resource_model_->isActive()) {
93 // ロードが終わっていない間は軽量モデルキャッシュ側を描画
94 auto* model_cache = resource_model_->modelCache();
95 model_cache->render(mat_world_);
96
97 return;
98 }
99
100 // ロード終了していたらハンドルを複製
101 on_initialize();
102
103 for(s32 mesh = 0; mesh < MV1GetMeshNum(mv1_handle_); ++mesh) { // モデルに含まれるメッシュの数
104 renderByMesh(mesh, override_vs, override_ps);
105 }
106}

108//---------------------------------------------------------------------------
109//! メッシュ番号指定で描画
110//---------------------------------------------------------------------------
111void Model::renderByMesh(s32 mesh, ShaderVs* override_vs, ShaderPs* override_ps) {
112 if(!resource_model_)
113 return;
114
115 if(!resource_model_->isActive())
116 return;
117
118 // ロード終了していたらハンドルを複製
119 on_initialize();
120
121 // メッシュ番号が有効範囲外
122 if(MV1GetMeshNum(mv1_handle_) <= mesh) {
123 return;
124 }
125
126 // ワールド行列を設定
127 MV1SetMatrix(mv1_handle_, mat_world_);
128
129 // シェーダーを使わない場合はDxLib関数を直接実行
130 if(!use_shader_) {
131 MV1DrawMesh(mv1_handle_, mesh);
132 return;
133 }
134
135 //--------------------------------------------------
136 // テクスチャ設定の上書き
137 //--------------------------------------------------
138 bool override_normalmap = false;
139
140 if(overridedTextures_[static_cast<s32>(Model::TextureType::Diffuse)]) {
141 SetUseTextureToShader(0, *overridedTextures_[static_cast<s32>(Model::TextureType::Diffuse)]);
142 }
143 if(overridedTextures_[static_cast<s32>(Model::TextureType::Normal)]) {
144 SetUseTextureToShader(1, *overridedTextures_[static_cast<s32>(Model::TextureType::Normal)]);
145 override_normalmap = true; // 法線マップを使用
146 }
147 if(overridedTextures_[static_cast<s32>(Model::TextureType::Specular)]) {
148 SetUseTextureToShader(2, *overridedTextures_[static_cast<s32>(Model::TextureType::Specular)]);
149 }
150
151 //--------------------------------------------------
152 // シェーダーで描画
153 //--------------------------------------------------
154
155 ShaderVs* vs = override_vs ? override_vs : shader_vs_.get();
156 ShaderPs* ps = override_ps ? override_ps : shader_ps_.get();
157
158 // オリジナルシェーダーを使用をONにする
159 MV1SetUseOrigShader(true);
160
161 SetUseTextureToShader(11, *textureIBL_diffuse_);
162 SetUseTextureToShader(12, *textureIBL_specular_);
163
164 for(s32 t = 0; t < MV1GetMeshTListNum(mv1_handle_, mesh); ++t) { // メッシュに含まれるトライアングルリストの数
165
166 // トライアングルリスト番号
167 auto tlist = MV1GetMeshTList(mv1_handle_, mesh, t);
168
169 // トライアングルリストが使用しているマテリアルのインデックスを取得する
170 auto material_index = MV1GetTriangleListUseMaterial(mv1_handle_, tlist);
171
172 // 法線マップを使用しているかどうか
173 bool use_normalmap = MV1GetMaterialNormalMapTexture(mv1_handle_, material_index) != -1;
174
175 // 法線マップを使用しない場合はNull法線を登録しておく
176 if(!use_normalmap && !override_normalmap) {
177 SetUseTextureToShader(1, *tex_null_normal_);
178 }
179
180 //--------------------------------------------------
181 // シェーダーバリエーションを選択
182 //--------------------------------------------------
183 // 頂点データタイプ(DX_MV1_VERTEX_TYPE_1FRAME 等)
184 auto vertex_type = MV1GetTriangleListVertexType(mv1_handle_, tlist);
185
186 u32 variant_vs = vertex_type; // DXライブラリの頂点タイプをそのままバリエーション番号に
187
188 //--------------------------------------------------
189 // トライアングルリストを描画
190 //--------------------------------------------------
191 int handle_vs = vs->variant(variant_vs);
192 int handle_ps = *ps;
193
194 // シェーダーがない場合はオリジナルシェーダー利用を無効化
195 bool shaderEnable = (handle_vs != -1) && (handle_ps != -1);
196 MV1SetUseOrigShader(shaderEnable);
197
198 // 頂点シェーダー
199 SetUseVertexShader(handle_vs);
200
201 // ピクセルシェーダー
202 SetUsePixelShader(handle_ps);
203
204 // 描画
205 MV1DrawTriangleList(mv1_handle_, tlist);
206 }
207
208 SetUseTextureToShader(11, -1);
209 SetUseTextureToShader(12, -1);
210
211 // オリジナルシェーダー使用をOFFにする
212 MV1SetUseOrigShader(false);
213
214 // 使い終わったらテクスチャ設定を解除
215 // 解除しておかないと他のモデル描画に影響あり。
216 SetUseTextureToShader(0, -1);
217 SetUseTextureToShader(1, -1);
218 SetUseTextureToShader(2, -1);
219}

236Model::~Model() {
237 // アニメーションが関連付けられている場合は相手の設定を解除
238 if(animation_) {
239 animation_->bindModel(nullptr);
240 }
241
242 // [DxLib] MV1モデルを解放 (複製されたハンドル)
243 if(mv1_handle_ != -1) {
244 MV1DeleteModel(mv1_handle_);
245 }
246
247 // 一番最後のオブジェクトが解放を担当
248 ref_counter_--;
249
250 if(ref_counter_ == 0) {
251 // シェーダー
252 shader_vs_.reset();
253 shader_ps_.reset();
254
255 // デフォルトテクスチャ
256 tex_null_white_.reset();
257 tex_null_black_.reset();
258 tex_null_normal_.reset();
259 }
260
261 if(textureIBL_diffuse_)
262 textureIBL_diffuse_.reset();
263 if(textureIBL_specular_)
264 textureIBL_specular_.reset();
265}

267//---------------------------------------------------------------------------
268//! アニメーションを設定
269//---------------------------------------------------------------------------
270void Model::bindAnimation(Animation* animation) {
271 if(animation == animation_)
272 return;
273
274 // 旧モデルを解除
275 auto* last = animation_;
276 if(last) {
277 animation_ = nullptr;
278 last->bindModel(nullptr);
279 }
280
281 // 新モデルに登録
282 animation_ = animation;
283 if(animation_) {
284 animation_->bindModel(this);
285 }
286}
