83void ComponentCamera::SetCameraTransform() {
84 if(camera_status_.is(CameraBit::ChangeReq)) {
85 auto current = current_camera_.lock();
86 // 現状は瞬時に切り替える
87 // TODO モーフィングなど
88 if(current)
89 current->camera_status_.off(CameraBit::Current);
90
91 camera_status_.on(CameraBit::Current);
92 camera_status_.off(CameraBit::ChangeReq);
93 current_camera_ = std::weak_ptr<ComponentCamera>(std::dynamic_pointer_cast<ComponentCamera>(shared_from_this()));
94 }
95
96 // if( !camera_status_.is( CameraBit::Current ) || ( DebugCamera::IsUse()
97 //&& !camera_status_.is( CameraBit::DebugCameara ) ) ) return;
98
99 auto position = GetPosition();
100 auto target = GetTarget();
101
102 if(!ImGuizmo::IsUsing()) {
103 mat_view_ = matrix::lookAtLH(position, target);
104 mat_proj_ = matrix::perspectiveFovLH(fovy_ * DegToRad, aspect_ratio_, near_z_, far_z_);
105 }
106
107 // デバッグカメラを使用してないとき、デバッグカメラで自分がデバッグカメラのときのみ有効とする
108 if(!DebugCamera::IsUse() || (DebugCamera::IsUse() && camera_status_.is(CameraBit::DebugCamera))) {
109 if(camera_status_.is(CameraBit::Current)) {
110 SetCameraViewMatrix(mat_view_); // ビュー行列
111 SetupCamera_ProjectionMatrix(mat_proj_); // 投影行列
112 }
113 }
114
115 // 視錐台(Frustum)の更新
116 if(camera_status_.is(CameraBit::Current)) {
117 frustum_.setPosition(position);
118 frustum_.setLookAt(target);
119 frustum_.setWorldUp(up_);
120 frustum_.setFov(fovy_ * DegToRad);
121 frustum_.setDepthMode(Frustum::DepthMode::Default);
122 frustum_.setAspectRatio(aspect_ratio_);
123
124 frustum_.setNearZ(near_z_);
125 frustum_.setFarZ(far_z_);
126 frustum_.update();
127 }
