50//! @brief ディレクトリパスのファイルを列挙します
51//! @param dirname ディレクトリパス
52//! @param ext 列挙する拡張子(ディフォルトは無視) 例:".png" ".jpg/.png/.bmp"
53//! など
54//! @return ディレクトリに存在するファイル
55Files_t GetFilesFromDirectory(std::string_view dirname, std::string_view ext) {
56 Files_t vfiles;
57 fs::recursive_directory_iterator files(dirname);
58
59 auto exts = String::Split(ext, '/');
60
61 for(const auto& f: files) {
62 auto path = f.path();
63 if(ext == "") {
64 vfiles.push_back(f.path().string());
65 continue;
66 }
67
68 for(auto e: exts) {
69 if(path.extension().string() == e) {
70 vfiles.push_back(f.path().string());
71 break;
72 }
73 }
74 }
75
76 return vfiles;
77}

239bool NearlyEqual(float f1, float f2) {
240 if(fabs(f1 - f2) <= FLT_EPSILON)
241 return true;
242

246matrix CreateMatrixByFrontVector(float3 front, float3 up, bool upremake) {
247 float3 n_front = normalize(front);
248 float3 n_up = normalize(up);
249 float3 n_right = normalize(cross(n_up, n_front));
250 float3 n_pos = {0, 0, 0};
251
252 if(upremake) {
253 n_up = normalize(cross(n_front, n_right));
254 }
255
256 matrix mat{
257 {n_right, 0}, //< X軸
258 { n_up, 0}, //< Y軸
259 {n_front, 0}, //< Z軸
260 { n_pos, 1}, //< POS
261 };
262 return mat;
263}

267 mat._41_42_43 = my_object_pos;
