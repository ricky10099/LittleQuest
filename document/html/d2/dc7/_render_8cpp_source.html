129//---------------------------------------------------------------------------
130//! RenderTargetを一括設定 (TargetDesc指定)
131//---------------------------------------------------------------------------
132void SetRenderTarget(const TargetDesc& desc) {
133 // 保存
134 target_desc_ = desc;
135
136 // カラーターゲットを配列として取得
137 std::array<ID3D11RenderTargetView*, D3D11_SIMULTANEOUS_RENDER_TARGET_COUNT> rtvs;
138 for(u32 i = 0; i < desc.color_count_; ++i) {
139 rtvs[i] = desc.color_targets_[i] ? desc.color_targets_[i]->rtv() : nullptr;
140 }
141
142 // デプスバッファ
143 ID3D11DepthStencilView* dsv = desc.depth_stencil_ ? desc.depth_stencil_->dsv() : nullptr;
144
145 // [DxLib] DXライブラリ内部のプリミティブバッファをフラッシュさせる
146 RenderVertex();
147
148 // RenderTarget設定
149 GetD3DDeviceContext()->OMSetRenderTargets(desc.color_count_, rtvs.data(), dsv);
150
151 //----------------------------------------------------------
152 // ビューポートとシザー領域を更新
153 //----------------------------------------------------------
154 {
155 // RenderTargetの解像度を取得
156 u32 w = desc.color_targets_[0]->width();
157 u32 h = desc.color_targets_[0]->height();
158
159 D3D11_VIEWPORT viewport{0, 0, static_cast<f32>(w), static_cast<f32>(h), 0.0f, 1.0f};
160 GetD3DDeviceContext()->RSSetViewports(1, &viewport);
161
162 D3D11_RECT rect{0, 0, static_cast<LONG>(w), static_cast<LONG>(h)};
163 GetD3DDeviceContext()->RSSetScissorRects(1, &rect);
164 }
165}

205//---------------------------------------------------------------------------
206//! RenderTargetにイメージをコピー
207//---------------------------------------------------------------------------
208void CopyToRenderTarget(Texture* dest_render_target, Texture* source_texture, int shader_ps_handle) {
209 // 今のRenderTarget情報を保存しておく(関数終了時に復元するため)
210 auto current_target_desc = GetRenderTarget();
211
212 // 描画先を変更
213 SetRenderTarget(dest_render_target);
214
215 SetDrawMode(DX_DRAWMODE_BILINEAR); // テクスチャをバイリニア補間
216 SetTextureAddressMode(DX_TEXADDRESS_CLAMP); // テクスチャを繰り返しなし
217
218 //----------------------------------------------------------
219 //! フルスクリーン2D三角形の描画
220 //! @see GDC'14 Vertex Shader Tricks
221 //! 通常の四角形描画では三角形2枚で描画しますが、斜めのエッジ部分が2回ラスタライズされます。
222 //! 無駄を省くために巨大な三角形で覆うことでこの問題を解決できます。(数%の高速化効果)
223 // A
224 // | ＼ 本来のスクリーンを△ABCで覆うと有効ピクセル部分は
225 // | ＼ 矩形ラスタライズされるようになります
226 // | ＼
227 // +------+------+
228 // | | | ＼
229 // +------+------+ ＼
230 // | | | ＼
231 // B-------------+-------------C
232 //----------------------------------------------------------
233 {
234 f32 w = static_cast<f32>(WINDOW_W); // 解像度 幅
235 f32 h = static_cast<f32>(WINDOW_H); // 解像度 高さ
236
237 VERTEX2DSHADER v[3]{};
238
239 // 頂点
240 v[0].pos = {0.0f, -h, 0.0f}; // 2D座標
241 v[0].rhw = 1.0f; // rhw = 1.0f 初期化は2D描画に必須
242 v[0].dif = GetColorU8(255, 255, 255, 255); // カラー
243 v[0].u = 0.0f; // テクスチャ座標 U
244 v[0].v = -1.0f; // テクスチャ座標 V
245
246 v[1].pos = {0.0f, h, 0.0f}; // 2D座標
247 v[1].rhw = 1.0f; // rhw = 1.0f 初期化は2D描画に必須
248 v[1].dif = GetColorU8(255, 255, 255, 255); // カラー
249 v[1].u = 0.0f; // テクスチャ座標 U
250 v[1].v = 1.0f; // テクスチャ座標 V
251
252 v[2].pos = {w * 2.0f, h, 0.0f}; // 2D座標
253 v[2].rhw = 1.0f; // rhw = 1.0f 初期化は2D描画に必須
254 v[2].dif = GetColorU8(255, 255, 255, 255); // カラー
255 v[2].u = 2.0f; // テクスチャ座標 U
256 v[2].v = 1.0f; // テクスチャ座標 V
257
258 // 使用するテクスチャを設定 (slot=0)
259 SetUseTextureToShader(0, *source_texture);
260
261 // 使用するピクセルシェーダーを設定 (2Dの場合は頂点シェーダー不要)
262 int ps = (shader_ps_handle != -1) ? shader_ps_handle : *shader_ps_texture_;
263 SetUsePixelShader(ps);
264
265 // 描画
266 DrawPrimitive2DToShader(v, 3, DX_PRIMTYPE_TRIANGLELIST);
267 }
268 // 使い終わったらテクスチャ設定を解除
269 // 解除しておかないとモデル描画に影響あり。
270 SetUseTextureToShader(0, -1);
271 SetUsePixelShader(-1);
272
273 // 描画先を元に戻す
274 SetRenderTarget(current_target_desc); // 元に戻す
275}
