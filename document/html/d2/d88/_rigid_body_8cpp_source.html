46class RigidBodyImpl: public physics::RigidBody {
47 public:
48 RigidBodyImpl(const JPH::ShapeRefC& shape, u16 layer, physics::MotionType motion_type) {
49 // ボディ自体の設定を作成します。ここで跳ね返り/摩擦係数のような他のプロパティも設定できます。
50 auto position = JPH::Vec3(0.0f, 0.0f, 0.0f);
51 auto rotation = JPH::Quat::sIdentity();
52
53 JPH::BodyCreationSettings body_settings(shape, // 形状
54 position, // 位置座標
55 rotation, // 回転姿勢
56 convertTo(motion_type), // 動的/静的/キネマティック
57 layer); // レイヤー番号
58
59 // CCD (Continuous Collision Detection) 有効
60 body_settings.mMotionQuality = JPH::EMotionQuality::LinearCast;
61
62 //------------------------------------------------------
63 // 剛体を作成
64 //------------------------------------------------------
65 // ボディを使い果たした場合はnullptrが返ります
66 jph_body_ = physics::Engine::bodyInterface()->CreateBody(body_settings);
67
68 if(jph_body_ == nullptr) {
69 assert(0 && "ボディ個数を最大数まで使い果たしました。");
70 return;
71 }
72
73 // ワールドに追加 (アクティブ化)
74 physics::Engine::bodyInterface()->AddBody(jph_body_->GetID(), JPH::EActivation::Activate);
75
76 // IDを保存
77 body_id_ = jph_body_->GetID();
78 }

428//---------------------------------------------------------------------------
429//! Mesh剛体を作成
430//---------------------------------------------------------------------------
431std::shared_ptr<physics::RigidBody> createRigidBody(const shape::Mesh& o, u16 layer) {
432 //----------------------------------------------------------
433 // 頂点配列を取得
434 //----------------------------------------------------------
435 JPH::VertexList vertexList;
436 JPH::IndexedTriangleList indexList;
437
438 for(auto& v: o.vertices_) {
439 JPH::Float3 position{v.x, v.y, v.z};
440 vertexList.emplace_back(std::move(position));
441 }
442 for(u32 i = 0; i < o.indices_.size(); i += 3) {
443 u32 i0 = o.indices_[i + 0];
444 u32 i1 = o.indices_[i + 1];
445 u32 i2 = o.indices_[i + 2];
446 JPH::IndexedTriangle triangle(i0, i1, i2, 0);
447 indexList.emplace_back(std::move(triangle));
448 }
449
450 JPH::PhysicsMaterialList materialList;
451 materialList.push_back(new JPH::PhysicsMaterialSimple("BPMeshMaterial", JPH::Color::sGetDistinctColor(0)));
452
453 //----------------------------------------------------------
454 // シェイプを生成
455 //----------------------------------------------------------
456 JPH::MeshShapeSettings shape_settings(vertexList, indexList, materialList);
457 JPH::ShapeSettings::ShapeResult result = shape_settings.Create();
458
459 if(result.HasError())
460 return nullptr;
461
462 // 作成
463 // メッシュ剛体は常に静的で生成されます
464 return std::make_shared<RigidBodyImpl>(result.Get(), layer, physics::MotionType::Static);
465}
