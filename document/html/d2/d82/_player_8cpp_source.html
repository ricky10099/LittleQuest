120void Player::GameAction() {
121 m_movement = float3(0, 0, 0);
122 m_hitTimer -= GetDeltaTime60();
123 m_hitTimer = std::max(0.0f, m_hitTimer);
124
125 if(m_hitTimer <= 0 && m_isHit) {
126 m_isHit = false;
127 }
128
129 if(!m_pCamera.lock()) {
130 m_pCamera = Scene::GetObjectPtr<Camera>("PlayerCamera");
131 m_cameraLength = m_pCamera.lock()->GetComponent<ComponentSpringArm>()->GetSpringArmLength();
132 } else {
133 float3 v = m_pCamera.lock()->CameraForward();
134 m_selfMatrix = HelperLib::Math::CreateMatrixByFrontVector(v);
135 }
136
137 InputHandle();
138
139 if(m_currCombo != Combo::NO_COMBO && m_playerState != PlayerState::ROLL) {
140 m_playerState = PlayerState::ATTACK;
141 }
142
143 if(IsFloat3Zero(m_movement) && m_playerState != PlayerState::GET_HIT && m_playerState != PlayerState::ATTACK &&
144 m_playerState != PlayerState::ROLL && m_playerState != PlayerState::DEAD) {
145 m_playerState = PlayerState::IDLE;
146 }
147
148 switch(m_playerState) {
149 case PlayerState::GET_HIT:
150 if(!m_pModel.lock()->IsPlaying()) {
151 m_playerState = PlayerState::IDLE;
152 }
153 break;
154 case PlayerState::ATTACK:
155 Attack();
156 break;
157 case PlayerState::ROLL:
158 m_pModel.lock()->PlayAnimationNoSame(STR(PlayerState::ROLL));
159 m_currCombo = Combo::NO_COMBO;
160 m_pWeapon.lock()->SetHitCollisionGroup((u32)ComponentCollision::CollisionGroup::NONE);
161 if(m_pModel.lock()->IsPlaying()) {
162 m_movement = {};
163 m_isInvincible = true;
164 float3 vec = m_pModel.lock()->GetMatrix().axisZ();
165 vec.y = 0;
166 m_movement += -vec;
167 m_movement *= 15.0f * GetDeltaTime60();
168 AddTranslate(m_movement);
169 } else {
170 m_isInvincible = false;
171 m_playerState = PlayerState::IDLE;
172 }
173 break;
174 case PlayerState::WALK:
175 Walk();
176 break;
177 case PlayerState::DEAD:
178 Die();
179 break;
180 default:
181 Idle();
182 break;
183 }

215void Player::OnHit([[maybe_unused]] const ComponentCollision::HitInfo& hitInfo) {
216 if((u32)hitInfo.collision_->GetCollisionGroup() & (u32)ComponentCollision::CollisionGroup::WEAPON) {
217 auto* owner = hitInfo.hit_collision_->GetOwner();
218
219 Boss* boss;
220 if((boss = dynamic_cast<Boss*>(owner)) && m_currCombo != Combo::NO_COMBO) {
221 bool inList = false;
222 for(int i = 0; i < m_attackList.size(); i++) {
223 if(m_attackList[i] == boss->GetName().data()) {
224 inList = true;
225 break;
226 }
227 }
228
229 if(!inList) {
230 m_attackList.push_back(boss->GetName().data());
231 boss->GetHit((int)(this->BASE_ATK * m_comboList[m_currCombo] * m_pCombo.lock()->ComboBuff()));
232 m_isHit = true;
233 m_hitTimer += HIT_PAUSE;
234 m_pCombo.lock()->AddCombo(m_comboList[m_currCombo]);

236 PlaySoundMem(m_swordHitSE, DX_PLAYTYPE_BACK);
237 ChangeVolumeSoundMem((int)(MAX_VOLUME * (Scene::GetSEVolume() / 100.0f)), m_swordHitSE);
238 SetPosPlayingEffekseer3DEffect(m_playingEffect, hitInfo.hit_position_.x, hitInfo.hit_position_.y,
239 hitInfo.hit_position_.z);
240 }
241 }

267void Player::InputHandle() {
268 DINPUT_JOYSTATE DInputState;
269 m_cameraLength -= GetMouseWheelRotVol() * 3;
270 if(IsPadRepeat(PAD_ID::PAD_5, DX_PADTYPE_DUAL_SENSE)) {
271 ++m_cameraLength;
272 }
273 if(IsPadRepeat(PAD_ID::PAD_6, DX_PADTYPE_DUAL_SENSE)) {
274 --m_cameraLength;
275 }
276 m_cameraLength = std::min((std::max(m_cameraLength, 10.0f)), 100.0f);
277 m_pCamera.lock()->SetCameraLength(m_cameraLength);
278 if(m_playerState != PlayerState::ROLL && m_playerState != PlayerState::GET_HIT && m_playerState != PlayerState::DEAD) {
279 if(IsKeyRepeat(KEY_INPUT_LSHIFT)) {
280 m_speedFactor = WALK_SPEED;
281 } else {
282 m_speedFactor = RUN_SPEED;
283 }
284
285 GetJoypadDirectInputState(DX_INPUT_PAD1, &DInputState);
286 switch(GetJoypadType(DX_INPUT_PAD1)) {
287 case DX_PADTYPE_DUAL_SENSE:
288 m_movement += float3{-DInputState.X, 0, DInputState.Y};
289 m_movement = mul(float4{m_movement, 1.0f}, m_selfMatrix).xyz;
290 break;
291 default:
292 break;
293 }
294
295 if(IsKeyRepeat(KEY_INPUT_W)) {
296 float3 vec = m_selfMatrix.axisZ();
297 vec.y = 0;
298 m_movement += -vec;
299 }
300 if(IsKeyRepeat(KEY_INPUT_D)) {
301 float3 vec = m_selfMatrix.axisX();
302 vec.y = 0;
303 m_movement += -vec;
304 }
305 if(IsKeyRepeat(KEY_INPUT_S)) {
306 float3 vec = m_selfMatrix.axisZ();
307 vec.y = 0;
308 m_movement += vec;
309 }
310 if(IsKeyRepeat(KEY_INPUT_A)) {
311 float3 vec = m_selfMatrix.axisX();
312 vec.y = 0;
313 m_movement += vec;
314 }
315
316 if(IsMouseDown(MOUSE_INPUT_LEFT) || IsPadDown(PAD_ID::PAD_4, DX_PADTYPE_DUAL_SENSE)) {
317 m_playerState = PlayerState::ATTACK;
318
319 if(m_currCombo == Combo::NO_COMBO) {
320 m_currCombo = Combo::NORMAL_COMBO1;
321 } else if(m_currCombo != Combo::SPECIAL_ATTACK) {
322 m_isCombo = true;
323 }
324 }
325
326 if(IsMouseRepeat(MOUSE_INPUT_RIGHT, 1) || IsPadRepeat(PAD_ID::PAD_3, DX_PADTYPE_DUAL_SENSE)) {
327 m_chargeTime += GetDeltaTime60();
328 if(m_chargeTime >= SPECIAL_CHARGE_TIME && !m_charged) {
329 StopEffekseer3DEffect(m_playingChargeEffect);
330 m_charged = true;
331 }
332 if(IsEffekseer3DEffectPlaying(m_playingChargeEffect) == -1) {
333 m_playingChargeEffect = PlayEffekseer3DEffect(m_pChargeList[(m_chargeTime >= SPECIAL_CHARGE_TIME)]);
334 }
335 SetPosPlayingEffekseer3DEffect(m_playingChargeEffect, GetTranslate().x, GetTranslate().y + 7, GetTranslate().z);
336 m_speedFactor = WALK_SPEED;
337 } else if(m_chargeTime > 0 && m_currCombo == Combo::NO_COMBO) {
338 StopEffekseer3DEffect(m_playingChargeEffect);
339 if(m_chargeTime >= SPECIAL_CHARGE_TIME) {
340 m_isCombo = true;
341 }
342 m_charged = false;
343 m_playerState = PlayerState::ATTACK;
344 m_currCombo = Combo::SPECIAL_ATTACK;
345 m_chargeTime = 0;
346 m_speedFactor = RUN_SPEED;
347 }
348
349 if(!IsFloat3Zero(m_movement)) {
350 m_playerState = PlayerState::WALK;
351 }
352 if(IsKeyDown(KEY_INPUT_SPACE) || IsPadDown(PAD_ID::PAD_2, DX_PADTYPE_DUAL_SENSE)) {
353 m_chargeTime = 0;
354 m_charged = false;
355 StopEffekseer3DEffect(m_playingChargeEffect);
356 m_playerState = PlayerState::ROLL;
357 this->SetModelRotation();
358 }
359 }

477void Player::AttackAnimation(std::string animName, AnimInfo animInfo, Combo nextCombo) {
478 if(m_pModel.lock()->GetPlayAnimationName() != animName) {
479 this->SetModelRotation();
480 m_pModel.lock()->PlayAnimationNoSame(animName, false, 0.2F, m_animList[animName].animStartTime);
481 m_pModel.lock()->SetAnimationSpeed(animInfo.animStartSpeed);
482 m_attackList.clear();
483 m_playedFX = false;
484 }
485 m_currAnimTime = m_pModel.lock()->GetAnimationPlayTime();
486 if(m_currAnimTime > m_animList[animName].triggerStartTime) {
487 if(m_isHit) {
488 m_pModel.lock()->SetAnimationSpeed(animInfo.animSpeed * 0.05f);
489 } else {
490 m_pModel.lock()->SetAnimationSpeed(animInfo.animSpeed);
491 }
492 m_pWeapon.lock()->SetHitCollisionGroup((u32)ComponentCollision::CollisionGroup::ENEMY);
493 }
494 if(m_currAnimTime > m_animList[animName].triggerEndTime) {
495 m_pWeapon.lock()->SetHitCollisionGroup((u32)ComponentCollision::CollisionGroup::NONE);
496 }
497 if(m_currAnimTime > m_animList[animName].animCutInTime) {
498 m_currCombo = Combo::NO_COMBO;
499 if(m_isCombo) {
500 m_currCombo = nextCombo;
501 m_isCombo = false;
502 }
503 }
