71//! @brief フレーム情報から姿勢(位置・回転・スケール)を取得
72//! @param frame フレーム
73matrix SequenceObject::GetTransformFromFrame(int frame) {
74 // frame からキー情報を取得するLambda関数
75 auto calc_value = [](int frame, std::vector<uint32_t> keys, std::vector<float3> values,
76 float3 default_value = {0, 0, 0}) -> float3 {
77 int old = -1;
78 float3 value = default_value;
79 for(int i = 0; i < keys.size(); ++i) {
80 int key = static_cast<int>(keys[i]);
81 if(key >= frame) {
82 float base = static_cast<float>(key - old);
83 float par = static_cast<float>(frame - old) / base;
84
85 value = value * (1.0f - par) + values[i] * par;
86 break;
87 } else {
88 old = key;
89 value = values[i];
90 }
91 }
92
93 return value;
94 };
95
96 // frameの時の位置の評価をする
97 float3 position = calc_value(frame, position_keys_, position_values_);
98 // frameの時の回転の評価をする
99 float3 rotation = calc_value(frame, rotation_keys_, rotation_values_);
100 // frameの時のスケールの評価をする
101 float3 scale = calc_value(frame, scale_keys_, scale_values_, {1, 1, 1});
102
103 matrix mat;
104 RecomposeMatrixFromComponents((float*)&position, (float*)&rotation, (float*)&scale, (float*)&mat);
105 return mat;
106}

139template<class T>
140void SortVectors(std::vector<uint32_t>& v1, std::vector<T>& v2) {
141 size_t n = v1.size();
142 std::vector<uint32_t> p(n), v1x(n);
143 std::vector<T> v2x(n);
144 std::iota(p.begin(), p.end(), 0);
145 std::sort(p.begin(), p.end(), [&](int a, int b) {
146 return v1[a] < v1[b];
147 });
148 for(int i = 0; i < n; i++) {
149 v1x[i] = v1[p[i]];
150 v2x[i] = v2[p[i]];
151 }
152 v1 = v1x;
153 v2 = v2x;
154}

196 component->Play(effect_loop_);

201 component->PlayPause( true );
202#endif

204 SetDeltaTime((frame - effect_frame_old_) / 60.0f);
205 component->Update();
206 SetDeltaTime(delta);
207 effect_frame_old_ = i_frame;
208 }
209 }
210 }
