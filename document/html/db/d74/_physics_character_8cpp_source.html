222//---------------------------------------------------------------------------
223//! コンストラクタ
224//---------------------------------------------------------------------------
225CharacterImpl::CharacterImpl(u16 layer) {
226 layer_ = layer;
227
228 //----------------------------------------------------------
229 // キャラクターコントローラーを作成
230 //----------------------------------------------------------
231 JPH::CharacterVirtualSettings settings;
232 {
233 // 坂道として認識する最大傾斜角
234 settings.mMaxSlopeAngle = 45.0f * DegToRad;
235
236 // キャラクターが他の物体を押すことができる最大の力（N）
237 settings.mMaxStrength = 2000.0f;
238
239 // ジオメトリからどれだけ離れられるかの距離
240 // これによってスイープができるだけ衝突しないようにして衝突コストを下げ、スタックするリスクを低減します。
241 // ぷるぷる振動しない程度に設定
242 settings.mCharacterPadding = 0.02f;
243
244 // 予測接触で形状の外側をどこまでスキャンするか
245 settings.mPredictiveContactDistance = 0.1f;
246
247 // めり込み状態から回復速度(0.0f～1.0f)
248 settings.mPenetrationRecoverySpeed = 1.0f;
249 }
250
251 // 作成
252 // (この段階ではまだシェイプは設定されていない)
253 jph_character_ =
254 new JPH::CharacterVirtual(&settings, JPH::Vec3::sZero(), JPH::Quat::sIdentity(), physics::Engine::physicsSystem());
255
256 // コンタクトリスナーを設定
257 // (このクラス自身がコンタクトリスナーを継承しています)
258 jph_character_->SetListener(this);
259}

370//---------------------------------------------------------------------------
371//! 移動
372//---------------------------------------------------------------------------
373float3 CharacterImpl::move(f32 delta_time, const float3& move_vector, const float3& old_position, const float3& jump_vector) {
374 float3 move_direction = move_vector;
375
376 //----------------------------------------------------------
377 // 階段歩行アルゴリズム
378 //----------------------------------------------------------
379 if constexpr(true) {
380 //------------------------------------------------------
381 // 水平方向にどれだけ移動させたいかを計算する
382 //------------------------------------------------------
383
384 // 現在の水平方向(XZ成分)の速度ベクトル
385 float3 current_horizontal_velocity = linearVelocity() * float3(1.0f, 0.0f, 1.0f) * delta_time;
386
387 // 水平方向のステップの長さ
388 f32 velocity_length = dot(current_horizontal_velocity, current_horizontal_velocity);
389 if(velocity_length != 0.0f) {
390 velocity_length = sqrtf(velocity_length);
391 }
392
393 // 実際に水平方向にどれだけ移動したかを計算する
394 float3 horizontal_step = (position() - old_position) * float3(1.0f, 0.0f, 1.0f);
395 f32 horizontal_length = dot(horizontal_step, horizontal_step);
396 if(horizontal_length != 0.0f) {
397 horizontal_length = sqrtf(horizontal_length);
398 }
399
400 // もし思うように進まず、急な斜面に面してしまった場合
401 if(horizontal_length + 0.0001f < velocity_length && canWalkStairs(move_direction)) {
402 // あとどれだけ前に出ればいいかを計算する
403 float3 step_dir = normalize(current_horizontal_velocity);
404 float3 step_forward = step_dir * (velocity_length - horizontal_length);
405
406 // 床をどこまで検査すればいいかを計算する
407 static constexpr f32 min_step_forward = 0.15f;
408 float3 step_forward_test = step_dir * min_step_forward;
409
410 static const float3 step_up_height = float3(0.0f, 0.4f, 0.0f); // 段差の高さ
411
412 // 階段を歩く
413 walkStairs(delta_time, step_up_height, step_forward, step_forward_test);
414 }
415 }
416
417 //------------------------------------------------------
418 // 斜面で滑っている時に法線と反対方向の動きをキャンセルする
419 //------------------------------------------------------
420 auto ground_state = groundState();
421 if(ground_state == physics::Character::GroundState::OnSteepGround) {
422 float3 normal = normalize(groundNormal() * float3(1.0f, 0.0f, 1.0f));
423
424 f32 dot_result = dot(normal, move_direction);
425 if(dot_result < 0.0f)
426 move_direction -= normal * dot_result;
427 }
428
429 //------------------------------------------------------
430 // Y成分の計算
431 //------------------------------------------------------
432 float3 new_velocity;
433
434 // 現在の垂直方向(Y成分)の速度ベクトル
435 float3 current_vertical_velocity = linearVelocity() * float3(0.0f, 1.0f, 0.0f);
436
437 // 移動床の速度
438 float3 ground_velocity = groundVelocity();
439
440 f32 relative_vertical_velocity = current_vertical_velocity.y - ground_velocity.y; // 相対速度
441
442 if(ground_state == physics::Character::GroundState::OnGround // 地面に接地している時
443 && (relative_vertical_velocity < 0.1f)) // かつ、地面から離れようとしていない時
444 {
445 // 移動床の速度を新しい速度にする
446 new_velocity = ground_velocity;
447
448 // ジャンプ
449 new_velocity += jump_vector; // ジャンプ速度
450 } else {
451 new_velocity = current_vertical_velocity;
452 }
453
454 return new_velocity;
455}

490//---------------------------------------------------------------------------
491//! キャラクターの形状を入れ替える（しゃがみ、構え方など）
492//---------------------------------------------------------------------------
493bool CharacterImpl::setShape(const shape::Base& shape, const float3& shape_offset) {
494 assert(shape.shapeType() == shape::Type::Capsule && "カプセル以外は未対応です.");
495
496 //------------------------------------------------------
497 // カプセルを生成
498 //------------------------------------------------------
499 std::unique_ptr<shape::Capsule> s = std::make_unique<shape::Capsule>(static_cast<const shape::Capsule&>(shape));
500 JPH::ShapeRefC jph_shape = new JPH::CapsuleShape(s->half_height_, s->radius_);
501
502 // 移動されたシェイプを作成
503 JPH::ShapeRefC translated_shape =
504 JPH::RotatedTranslatedShapeSettings(castJPH(shape_offset), JPH::Quat::sIdentity(), jph_shape).Create().Get();
505
506 //------------------------------------------------------
507 // シェイプを切り替え
508 // 形状を切り替える前に新しい形状が衝突するかどうかを確認します
509 //------------------------------------------------------
510 auto* physics_system = physics::Engine::physicsSystem();
511
512 auto broad_phase_layer_filter = physics_system->GetDefaultBroadPhaseLayerFilter(layer_);
513 auto layer_filter = physics_system->GetDefaultLayerFilter(layer_);
514
515 bool is_succeed = jph_character_->SetShape(translated_shape, // シェイプ
516 0.1f, // 切り替え後に許容される最大めり込み量
517 broad_phase_layer_filter, layer_filter, JPH::BodyFilter{}, JPH::ShapeFilter{},
518 *physics::Engine::tempAllocator());
519
520 if(is_succeed) {
521 // 記録内容を更新
522 jph_shape_ = jph_shape;
523 shape_ = std::move(s); // 形状
524 shape_offset_ = shape_offset; // シェイプの移動オフセット
525 }
526 return is_succeed;
527}

586//---------------------------------------------------------------------------
587//! キャラクターが指定されたボディと衝突可能かどうかをチェックします。
588//---------------------------------------------------------------------------
589bool CharacterImpl::OnContactValidate([[maybe_unused]] const JPH::CharacterVirtual* character, const JPH::BodyID& body_id2,
