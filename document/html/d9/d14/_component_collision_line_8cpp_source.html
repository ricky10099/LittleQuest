183//! @brief ワールドMatrixの取得
184//! @return 他のコンポーネントも含めた位置
185const matrix ComponentCollisionLine::GetWorldMatrix() const {
186 matrix transform = collision_transform_;
187
188 auto obj = GetOwner();
189 if(attach_node_ >= 0) {
190 auto mdl = obj->GetComponent<ComponentModel>();
191 if(mdl) {
192 transform = mul(transform, attach_node_matrix_);
193 }
194 } else {
195 matrix mat = obj->GetMatrix();
196 transform = mul(transform, mat);
197 }
198
199 return transform;
200}
